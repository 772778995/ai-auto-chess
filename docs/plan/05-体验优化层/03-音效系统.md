# 05.3 音效系统

## 目标

实现战斗音效、UI交互音效、背景音乐。

## 实现代码

```typescript
// packages/web/src/composables/useAudio.ts

import { ref } from 'vue'

const sounds: Map<string, HTMLAudioElement> = new Map()

export function useAudio() {
  const isMuted = ref(false)
  const bgmVolume = ref(0.5)
  const sfxVolume = ref(0.7)
  
  // 预加载音效
  function preloadSounds() {
    const soundList = [
      'attack',
      'hit',
      'death',
      'buy',
      'sell',
      'refresh',
      'win',
      'lose',
      'dice-roll'
    ]
    
    for (const name of soundList) {
      const audio = new Audio(`/assets/sounds/${name}.mp3`)
      sounds.set(name, audio)
    }
  }
  
  // 播放音效
  function playSfx(name: string) {
    if (isMuted.value) return
    
    const sound = sounds.get(name)
    if (sound) {
      sound.volume = sfxVolume.value
      sound.currentTime = 0
      sound.play()
    }
  }
  
  // 播放背景音乐
  function playBgm(name: string) {
    const bgm = sounds.get(`bgm-${name}`)
    if (bgm) {
      bgm.volume = bgmVolume.value
      bgm.loop = true
      bgm.play()
    }
  }
  
  // 静音切换
  function toggleMute() {
    isMuted.value = !isMuted.value
  }
  
  return {
    isMuted,
    bgmVolume,
    sfxVolume,
    preloadSounds,
    playSfx,
    playBgm,
    toggleMute,
  }
}
```

## 音效列表

| 类别 | 音效 | 触发时机 |
|------|------|----------|
| 战斗 | attack | 随从攻击 |
| 战斗 | hit | 受到伤害 |
| 战斗 | death | 随从死亡 |
| 战斗 | win/lose | 战斗结束 |
| UI | buy | 购买随从 |
| UI | sell | 卖出随从 |
| UI | refresh | 刷新商店 |
| UI | hover | 鼠标悬停 |

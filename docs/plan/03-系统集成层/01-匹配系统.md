# 03.1 匹配系统

## 目标

实现 8 人在线匹配，包括房间创建、加入、匹配队列。

## 匹配流程

```
玩家A              匹配队列              游戏房间
  │                   │                   │
  │  开始匹配         │                   │
  ├──────────────────>│                   │
  │                   │  加入队列         │
  │                   ├──────────────────>│
  │                   │                   │
  │                   │  队列满8人?       │
  │                   │<──────────────────┤
  │                   │                   │
  │                   │  创建房间         │
  │                   ├──────────────────>│
  │  匹配成功         │                   │
  │<──────────────────┤                   │
  │                   │                   │
```

## 1. 匹配服务 (matchService.ts)

```typescript
// packages/server/src/services/matchService.ts

import { EventEmitter } from 'events'
import { roomService } from './roomService'
import { gameService } from './gameService'
import type { User } from '../db/schema'

interface QueuedPlayer {
  userId: string
  username: string
  joinedAt: number
  socketId: string
}

interface MatchRoom {
  id: string
  players: QueuedPlayer[]
  createdAt: number
}

export class MatchService extends EventEmitter {
  private queue: QueuedPlayer[] = []
  private activeRooms: Map<string, MatchRoom> = new Map()
  private readonly MIN_PLAYERS = 2
  private readonly MAX_PLAYERS = 8
  private readonly MATCH_TIMEOUT = 30000 // 30秒
  private matchTimer: NodeJS.Timeout | null = null

  // 加入匹配队列
  async joinQueue(player: QueuedPlayer): Promise<boolean> {
    // 检查是否已在队列
    if (this.queue.some(p => p.userId === player.userId)) {
      return false
    }

    this.queue.push(player)
    this.emit('playerJoined', player)

    // 检查是否可以开始匹配
    this.checkMatchCondition()

    return true
  }

  // 离开匹配队列
  leaveQueue(userId: string): boolean {
    const index = this.queue.findIndex(p => p.userId === userId)
    if (index === -1) return false

    const player = this.queue.splice(index, 1)[0]
    this.emit('playerLeft', player)

    return true
  }

  // 检查匹配条件
  private checkMatchCondition(): void {
    // 满8人立即匹配
    if (this.queue.length >= this.MAX_PLAYERS) {
      this.createMatch(this.MAX_PLAYERS)
      return
    }

    // 启动匹配倒计时
    if (this.queue.length >= this.MIN_PLAYERS && !this.matchTimer) {
      this.matchTimer = setTimeout(() => {
        this.matchTimer = null
        if (this.queue.length >= this.MIN_PLAYERS) {
          this.createMatch(this.queue.length)
        }
      }, this.MATCH_TIMEOUT)
    }
  }

  // 创建匹配
  private async createMatch(playerCount: number): void {
    // 取消倒计时
    if (this.matchTimer) {
      clearTimeout(this.matchTimer)
      this.matchTimer = null
    }

    // 取出玩家
    const players = this.queue.splice(0, playerCount)

    // 创建房间
    const host = players[0]
    const room = await roomService.createRoom(
      host.userId,
      `匹配房间-${Date.now()}`,
      this.MAX_PLAYERS
    )

    // 加入其他玩家
    for (const player of players.slice(1)) {
      await roomService.joinRoom(room.id, player.userId)
    }

    // 全部准备
    for (const player of players) {
      await roomService.setReady(room.id, player.userId, true)
    }

    // 创建匹配房间记录
    const matchRoom: MatchRoom = {
      id: room.id,
      players,
      createdAt: Date.now(),
    }
    this.activeRooms.set(room.id, matchRoom)

    // 通知所有玩家
    this.emit('matchCreated', {
      roomId: room.id,
      players: players.map(p => ({
        userId: p.userId,
        username: p.username,
      })),
    })
  }

  // 获取队列信息
  getQueueInfo(): { count: number; players: QueuedPlayer[] } {
    return {
      count: this.queue.length,
      players: [...this.queue],
    }
  }

  // 取消匹配
  async cancelMatch(roomId: string, userId: string): Promise<boolean> {
    const matchRoom = this.activeRooms.get(roomId)
    if (!matchRoom) return false

    // 只有房主可以取消
    if (matchRoom.players[0].userId !== userId) return false

    // 解散房间
    await roomService.deleteRoom(roomId)
    this.activeRooms.delete(roomId)

    // 返回玩家到队列或通知
    this.emit('matchCancelled', { roomId })

    return true
  }
}

export const matchService = new MatchService()
```

## 2. 匹配 API

```typescript
// packages/server/src/routes/api/match.ts

import { Hono } from 'hono'
import { matchService } from '../../services/matchService'
import { authMiddleware } from '../../middleware/auth'

const match = new Hono()

// 加入匹配队列
match.post('/join', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const username = c.get('username')
  const { socketId } = await c.req.json()

  const success = await matchService.joinQueue({
    userId,
    username,
    socketId,
    joinedAt: Date.now(),
  })

  if (!success) {
    return c.json({
      success: false,
      error: 'Already in queue',
      timestamp: Date.now(),
    }, 400)
  }

  return c.json({
    success: true,
    data: { message: 'Joined queue' },
    timestamp: Date.now(),
  })
})

// 离开匹配队列
match.post('/leave', authMiddleware, async (c) => {
  const userId = c.get('userId')

  const success = matchService.leaveQueue(userId)

  return c.json({
    success,
    data: { message: success ? 'Left queue' : 'Not in queue' },
    timestamp: Date.now(),
  })
})

// 获取队列状态
match.get('/queue', async (c) => {
  const info = matchService.getQueueInfo()

  return c.json({
    success: true,
    data: info,
    timestamp: Date.now(),
  })
})

// 取消匹配
match.post('/cancel/:roomId', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const roomId = c.req.param('roomId')

  const success = await matchService.cancelMatch(roomId, userId)

  return c.json({
    success,
    data: { message: success ? 'Match cancelled' : 'Cannot cancel' },
    timestamp: Date.now(),
  })
})

export default match
```

## 3. 匹配状态管理 (matchStore.ts)

```typescript
// packages/web/src/stores/match.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useWebSocket } from '@/composables/useWebSocket'

export const useMatchStore = defineStore('match', () => {
  // ==================== State ====================
  
  const isMatching = ref(false)
  const queuePosition = ref(0)
  const queueCount = ref(0)
  const estimatedTime = ref(0)
  const matchedRoomId = ref<string | null>(null)
  const matchFound = ref(false)

  // ==================== Getters ====================
  
  const canStartMatch = computed(() => !isMatching.value && !matchFound.value)

  // ==================== Actions ====================
  
  const { sendMessage, onMessage } = useWebSocket()

  // 开始匹配
  async function startMatch(): Promise<boolean> {
    if (!canStartMatch.value) return false

    try {
      const response = await fetch('/api/match/join', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ socketId: getSocketId() }),
      })

      const result = await response.json()
      
      if (result.success) {
        isMatching.value = true
        startQueuePolling()
        return true
      }
      
      return false
    } catch (error) {
      console.error('Failed to start match:', error)
      return false
    }
  }

  // 取消匹配
  async function cancelMatch(): Promise<void> {
    if (!isMatching.value) return

    try {
      await fetch('/api/match/leave', { method: 'POST' })
      isMatching.value = false
      stopQueuePolling()
    } catch (error) {
      console.error('Failed to cancel match:', error)
    }
  }

  // 轮询队列状态
  let pollingInterval: number | null = null

  function startQueuePolling(): void {
    pollingInterval = window.setInterval(async () => {
      try {
        const response = await fetch('/api/match/queue')
        const result = await response.json()
        
        if (result.success) {
          queueCount.value = result.data.count
          // 计算预计等待时间
          estimatedTime.value = Math.max(0, (8 - queueCount.value) * 5)
        }
      } catch (error) {
        console.error('Queue polling error:', error)
      }
    }, 1000)
  }

  function stopQueuePolling(): void {
    if (pollingInterval) {
      clearInterval(pollingInterval)
      pollingInterval = null
    }
  }

  // 监听 WebSocket 匹配事件
  onMessage('match:found', (data) => {
    matchFound.value = true
    matchedRoomId.value = data.roomId
    isMatching.value = false
    stopQueuePolling()
  })

  onMessage('match:cancelled', () => {
    matchFound.value = false
    matchedRoomId.value = null
  })

  // 接受匹配
  async function acceptMatch(): Promise<void> {
    if (!matchedRoomId.value) return

    // 进入房间
    // ...
  }

  return {
    isMatching,
    queuePosition,
    queueCount,
    estimatedTime,
    matchedRoomId,
    matchFound,
    canStartMatch,
    startMatch,
    cancelMatch,
    acceptMatch,
  }
})
```

## 4. 匹配界面 (MatchmakingPanel.vue)

```vue
<template>
  <div _bg="dark-800" _rounded="lg" _p="6" _text="center">

      <p _text="gray-400" _mb="6">寻找最多7名其他玩家</p>
```

# 01.2 数据文件

## 目标

创建游戏数据模板，为后续开发提供测试数据。

## 文件结构

```
packages/shared/src/data/
├── followers.ts          # 随从数据

├── spells.ts             # 咒术数据
├── heroes.ts             # 英雄数据
└── index.ts              # 导出入口
```

export function getFollowerById(id: string): Follower | undefined {
  return followers.find(f => f.id === id)
}

export function getFollowersByLevel(level: number): Follower[] {
  return followers.filter(f => f.level === level)
}

export function getRandomFollowers(count: number, maxLevel: number): Follower[] {
  const available = followers.filter(f => f.level <= maxLevel)
  const result: Follower[] = []
  for (let i = 0; i < count; i++) {
    const randomIndex = Math.floor(Math.random() * available.length)
    result.push(available[randomIndex])
  }
  return result
}

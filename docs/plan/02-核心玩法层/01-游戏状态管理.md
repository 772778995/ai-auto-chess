# 02.1 游戏状态管理

## 目标

创建 Pinia Store，管理游戏全局状态。

## 文件结构

```
packages/web/src/stores/
├── game.ts              # 主游戏状态管理
├── shop.ts              # 商店状态（可合并到game.ts）
```

## 1. 主游戏 Store (game.ts)

```typescript
// packages/web/src/stores/game.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type {
  GamePhase,
  RecruitSubPhase,
  CombatSubPhase,
  Player,
  FollowerInstance,
  GameSession,
} from '@shared/types'

export const useGameStore = defineStore('game', () => {
  // ==================== State ====================
  
  // 游戏会话信息
  const sessionId = ref<string>('')
  const roomId = ref<string>('')
  const currentRound = ref<number>(1)
  const phase = ref<GamePhase>('lobby')
  const subPhase = ref<RecruitSubPhase | CombatSubPhase>()
  const turnTimer = ref<number>(60)
  
  // 玩家信息
  const currentPlayer = ref<Player | null>(null)
  const opponentPreview = ref<Player | null>(null)
  const allPlayers = ref<Player[]>([])
  
  // 游戏状态
  const isMyTurn = ref(false)
  const isCombatInProgress = ref(false)
  const lastCombatResult = ref<CombatResult | null>(null)
  
  // ==================== Getters ====================
  
  const isRecruitPhase = computed(() => phase.value === 'recruit')
  const isCombatPhase = computed(() => phase.value === 'combat')
  const isGameEnded = computed(() => phase.value === 'ended')
  
  const canShop = computed(() => 
    phase.value === 'recruit' && subPhase.value === 'shopping'
  )
  
  const canPosition = computed(() =>
    phase.value === 'recruit' && subPhase.value === 'positioning'
  )
  
  // 计算可放置随从的空位
  const availableBattleSlots = computed(() => {
    if (!currentPlayer.value) return []
    const slots: number[] = []
    currentPlayer.value.battlefield.forEach((follower, index) => {
      if (!follower) slots.push(index)
    })
    return slots
  })
  
  // 计算战斗力（简单估算）
  const combatPower = computed(() => {
    if (!currentPlayer.value) return 0
    return currentPlayer.value.battlefield.reduce((total, follower) => {
      if (!follower) return total
      return total + follower.currentAttack + follower.currentHealth
    }, 0)
  })
  
  // ==================== Actions ====================
  
  // 初始化游戏
  function initGame(session: GameSession, player: Player) {
    sessionId.value = session.id
    roomId.value = session.roomId
    currentRound.value = session.currentRound
    phase.value = session.phase
    subPhase.value = session.subPhase
    turnTimer.value = session.turnTimer
    currentPlayer.value = player
  }
  
  // 更新游戏阶段
  function updatePhase(
    newPhase: GamePhase,
    newSubPhase?: RecruitSubPhase | CombatSubPhase
  ) {
    phase.value = newPhase
    subPhase.value = newSubPhase
    
    // 阶段切换时的逻辑
    if (newPhase === 'recruit') {
      // 招募阶段开始，重置计时器
      turnTimer.value = 60
      isCombatInProgress.value = false
    } else if (newPhase === 'combat') {
      isCombatInProgress.value = true
    }
  }
  
  // 更新回合数
  function nextRound() {
    currentRound.value++
  }
  
  // 更新玩家状态
  function updatePlayer(updates: Partial<Player>) {
    if (!currentPlayer.value) return
    currentPlayer.value = { ...currentPlayer.value, ...updates }
  }
  
  // 更新金币
  function setGold(amount: number) {
    if (!currentPlayer.value) return
    currentPlayer.value.gold = amount
  }
  
  function addGold(amount: number) {
    if (!currentPlayer.value) return
    currentPlayer.value.gold += amount
  }
  
  // 更新生命
  function takeDamage(amount: number) {
    if (!currentPlayer.value) return
    currentPlayer.value.health = Math.max(0, currentPlayer.value.health - amount)
    if (currentPlayer.value.health <= 0) {
      currentPlayer.value.isAlive = false
    }
  }
  
  // 更新经验
  function addExperience(amount: number) {
    if (!currentPlayer.value) return
    currentPlayer.value.experience += amount
    checkLevelUp()
  }
  
  // 检查升级
  function checkLevelUp() {
    if (!currentPlayer.value) return
    const needed = currentPlayer.value.experienceToNext
    if (currentPlayer.value.experience >= needed) {
      currentPlayer.value.level++
      currentPlayer.value.experience -= needed
      currentPlayer.value.experienceToNext = calculateNextLevelExp(currentPlayer.value.level)
    }
  }
  
  // 计算升级所需经验
  function calculateNextLevelExp(level: number): number {
    // 等级2需要2点，等级3需要4点，等级4需要6点...
    return level * 2
  }
  
  // 设置准备状态
  function setReady(ready: boolean) {
    // 通过 WebSocket 发送准备状态
  }
  
  // 设置战斗结果
  function setCombatResult(result: CombatResult) {
    lastCombatResult.value = result
    isCombatInProgress.value = false
  }
  
  // 更新对手预览
  function updateOpponentPreview(opponent: Player) {
    opponentPreview.value = opponent
  }
  
  // 重置游戏
  function resetGame() {
    sessionId.value = ''
    roomId.value = ''
    currentRound.value = 1
    phase.value = 'lobby'
    subPhase.value = undefined
    turnTimer.value = 60
    currentPlayer.value = null
    opponentPreview.value = null
    allPlayers.value = []
    isMyTurn.value = false
    isCombatInProgress.value = false
    lastCombatResult.value = null
  }
  
  return {
    // State
    sessionId,
    roomId,
    currentRound,
    phase,
    subPhase,
    turnTimer,
    currentPlayer,
    opponentPreview,
    allPlayers,
    isMyTurn,
    isCombatInProgress,
    lastCombatResult,
    
    // Getters
    isRecruitPhase,
    isCombatPhase,
    isGameEnded,
    canShop,
    canPosition,
    availableBattleSlots,
    combatPower,
    
    // Actions
    initGame,
    updatePhase,
    nextRound,
    updatePlayer,
    setGold,
    addGold,
    takeDamage,
    addExperience,
    checkLevelUp,
    setReady,
    setCombatResult,
    updateOpponentPreview,
    resetGame,
  }
})
```

## 2. 商店 Store (shop.ts)

```typescript
// packages/web/src/stores/shop.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Follower, Spell, Equipment } from '@shared/types'
import { useGameStore } from './game'

// 随从刷新概率表
const REFRESH_RATES: Record<number, number[]> = {
  1: [1.0, 0, 0, 0, 0, 0],      // 等级1: 100% 1星
  2: [0.7, 0.3, 0, 0, 0, 0],    // 等级2: 70% 1星, 30% 2星
  3: [0.5, 0.4, 0.1, 0, 0, 0],  // 等级3: 50% 1星, 40% 2星, 10% 3星
  4: [0.35, 0.4, 0.2, 0.05, 0, 0],
  5: [0.25, 0.35, 0.25, 0.12, 0.03, 0],
  6: [0.18, 0.28, 0.28, 0.18, 0.07, 0.01],
}

export const useShopStore = defineStore('shop', () => {
  const gameStore = useGameStore()
  
  // ==================== State ====================
  
  const shopSlots = ref<(Follower | null)[]>([null, null, null, null, null])
  const isLocked = ref(false)
  const refreshCost = 1
  
  // ==================== Getters ====================
  
  const currentLevel = computed(() => gameStore.currentPlayer?.level || 1)
  
  const canRefresh = computed(() => {
    return (gameStore.currentPlayer?.gold || 0) >= refreshCost
  })
  
  // ==================== Actions ====================
  
  // 刷新商店
  function refreshShop() {
    if (isLocked.value) return
    if (!canRefresh.value) return
    
    // 扣除金币
    gameStore.addGold(-refreshCost)
    
    // 根据等级生成随从
    const rates = REFRESH_RATES[currentLevel.value]
    const newShop: (Follower | null)[] = []
    
    for (let i = 0; i < 5; i++) {
      const follower = generateRandomFollower(rates)
      newShop.push(follower)
    }
    
    shopSlots.value = newShop
  }
  
  // 锁定/解锁商店
  function toggleLock() {
    isLocked.value = !isLocked.value
  }
  
  // 购买随从
  function buyFollower(slotIndex: number): boolean {
    const follower = shopSlots.value[slotIndex]
    if (!follower) return false
    
    const cost = getFollowerCost(follower.level)
    if ((gameStore.currentPlayer?.gold || 0) < cost) return false
    
    // 检查手牌空间
    if ((gameStore.currentPlayer?.hand.length || 0) >= 10) return false
    
    // 扣除金币
    gameStore.addGold(-cost)
    
    // 添加到手牌
    const instance = createFollowerInstance(follower, gameStore.currentPlayer!.id)
    gameStore.currentPlayer!.hand.push(instance)
    
    // 清空该槽位
    shopSlots.value[slotIndex] = null
    
    return true
  }
  
  // 卖出随从
  function sellFollower(instanceId: string, fromHand: boolean = true): boolean {
    const player = gameStore.currentPlayer
    if (!player) return false
    
    let follower: FollowerInstance | undefined
    
    if (fromHand) {
      const index = player.hand.findIndex(f => f.instanceId === instanceId)
      if (index === -1) return false
      follower = player.hand[index]
      player.hand.splice(index, 1)
    } else {
      const index = player.battlefield.findIndex(f => f?.instanceId === instanceId)
      if (index === -1) return false
      follower = player.battlefield[index]!
      player.battlefield[index] = null
    }
    
    // 返还金币（等级决定）
    const sellPrice = getSellPrice(follower.level)
    gameStore.addGold(sellPrice)
    
    // 返还装备（如果有）
    if (follower.equipment && follower.equipment.length > 0) {
      // 装备回到玩家库存或商店
    }
    
    return true
  }
  
  // 生成随机随从
  function generateRandomFollower(rates: number[]): Follower | null {
    const random = Math.random()
    let cumulative = 0
    let targetLevel = 1
    
    for (let i = 0; i < rates.length; i++) {
      cumulative += rates[i]
      if (random <= cumulative) {
        targetLevel = i + 1
        break
      }
    }
    
    // 从数据中随机获取对应等级的随从
    const candidates = getFollowersByLevel(targetLevel)
    if (candidates.length === 0) return null
    
    return candidates[Math.floor(Math.random() * candidates.length)]
  }
  
  // 获取随从价格
  function getFollowerCost(level: number): number {
    return 3 // 所有随从都是3金币
  }
  
  // 获取卖出价格
  function getSellPrice(level: number): number {
    return 1 // 卖出获得1金币
  }
  
  // 创建随从实例
  function createFollowerInstance(follower: Follower, ownerId: string): FollowerInstance {
    return {
      ...follower,
      instanceId: generateInstanceId(),
      ownerId,
      position: { side: 'ally', x: 0, y: 0 },
      currentAttack: follower.attack,
      currentHealth: follower.health,
      currentShield: 0,
      equipment: [],
      buffs: [],
      debuffs: [],
    }
  }
  
  // 生成实例ID
  function generateInstanceId(): string {
    return `inst_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`
  }
  
  return {
    shopSlots,
    isLocked,
    refreshCost,
    canRefresh,
    refreshShop,
    toggleLock,
    buyFollower,
    sellFollower,
  }
})
```

## 3. 战场 Store (battlefield.ts)

```typescript
// packages/web/src/stores/battlefield.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { FollowerInstance, BattlePosition } from '@shared/types'
import { useGameStore } from './game'

export const useBattlefieldStore = defineStore('battlefield', () => {
  const gameStore = useGameStore()
  
  // ==================== State ====================
  
  // 拖拽状态
  const draggedFollower = ref<FollowerInstance | null>(null)
  const dragSource = ref<'hand' | 'battlefield' | null>(null)
  const dragSourceIndex = ref<number>(-1)
  
  // 选中状态
  const selectedFollower = ref<FollowerInstance | null>(null)
  const selectedSlot = ref<number>(-1)
  
  // ==================== Getters ====================
  
  const battlefield = computed(() => gameStore.currentPlayer?.battlefield || [])
  const hand = computed(() => gameStore.currentPlayer?.hand || [])
  
  const canPlaceFollower = computed(() => {
    return (slotIndex: number) => {
      // 只能在招募阶段且有空位时放置
      if (!gameStore.canPosition) return false
      return battlefield.value[slotIndex] === null
    }
  })
  
  // ==================== Actions ====================
  
  // 开始拖拽（从手牌）
  function startDragFromHand(follower: FollowerInstance, handIndex: number) {
    draggedFollower.value = follower
    dragSource.value = 'hand'
    dragSourceIndex.value = handIndex
  }
  
  // 开始拖拽（从战场）
  function startDragFromBattlefield(follower: FollowerInstance, slotIndex: number) {
    draggedFollower.value = follower
    dragSource.value = 'battlefield'
    dragSourceIndex.value = slotIndex
  }
  
  // 放置到战场
  function placeOnBattlefield(targetSlot: number): boolean {
    if (!draggedFollower.value) return false
    if (!canPlaceFollower.value(targetSlot)) return false
    
    const player = gameStore.currentPlayer
    if (!player) return false
    
    if (dragSource.value === 'hand') {
      // 从手牌放置
      player.hand.splice(dragSourceIndex.value, 1)
      player.battlefield[targetSlot] = draggedFollower.value
      
      // 更新位置
      draggedFollower.value.position = {
        side: 'ally',
        x: targetSlot % 3,
        y: Math.floor(targetSlot / 3),
      }
      
      // 触发入场效果
      triggerEnterEffect(draggedFollower.value)
      
    } else if (dragSource.value === 'battlefield') {
      // 战场内交换位置
      const sourceSlot = dragSourceIndex.value
      if (sourceSlot === targetSlot) return false
      
      // 交换
      const temp = player.battlefield[targetSlot]
      player.battlefield[targetSlot] = player.battlefield[sourceSlot]
      player.battlefield[sourceSlot] = temp
      
      // 更新位置
      if (player.battlefield[targetSlot]) {
        player.battlefield[targetSlot]!.position = {
          side: 'ally',
          x: targetSlot % 3,
          y: Math.floor(targetSlot / 3),
        }
      }
    }
    
    clearDrag()
    return true
  }
  
  // 移回手牌
  function returnToHand(slotIndex: number): boolean {
    const player = gameStore.currentPlayer
    if (!player) return false
    
    const follower = player.battlefield[slotIndex]
    if (!follower) return false
    
    // 检查手牌空间
    if (player.hand.length >= 10) return false
    
    // 移回手牌
    player.hand.push(follower)
    player.battlefield[slotIndex] = null
    
    // 清除位置
    follower.position = { side: 'ally', x: 0, y: 0 }
    
    return true
  }
  
  // 选择随从查看详情
  function selectFollower(follower: FollowerInstance | null) {
    selectedFollower.value = follower
  }
  
  // 装备物品
  function equipItem(followerInstanceId: string, equipment: Equipment): boolean {
    const player = gameStore.currentPlayer
    if (!player) return false
    
    // 查找随从
    let follower: FollowerInstance | undefined
    let slotIndex = -1
    
    for (let i = 0; i < player.battlefield.length; i++) {
      if (player.battlefield[i]?.instanceId === followerInstanceId) {
        follower = player.battlefield[i]!
        slotIndex = i
        break
      }
    }
    
    if (!follower) return false
    
    // 检查装备槽
    if (follower.equipment.length >= follower.equipmentSlots) {
      // 装备槽已满，需要替换
      return false
    }
    
    // 添加装备
    follower.equipment.push(equipment)
    
    // 应用装备效果
    applyEquipmentEffects(follower, equipment)
    
    return true
  }
  
  // 触发入场效果
  function triggerEnterEffect(follower: FollowerInstance) {
    const enterKeywords = follower.keywords.filter(
      k => k.trigger === TriggerTiming.ON_ENTER
    )
    
    for (const keyword of enterKeywords) {
      executeEffect(keyword.effect, follower)
    }
  }
  
  // 应用装备效果
  function applyEquipmentEffects(follower: FollowerInstance, equipment: Equipment) {
    for (const effect of equipment.effects) {
      if (effect.type === 'stat_boost') {
        if (effect.stat === 'attack') {
          follower.currentAttack += effect.value || 0
        } else if (effect.stat === 'health') {
          follower.currentHealth += effect.value || 0
        } else if (effect.stat === 'shield') {
          follower.currentShield += effect.value || 0
        }
      }
      // 其他效果类型...
    }
  }
  
  // 清除拖拽状态
  function clearDrag() {
    draggedFollower.value = null
    dragSource.value = null
    dragSourceIndex.value = -1
  }
  
  return {
    draggedFollower,
    dragSource,
    dragSourceIndex,
    selectedFollower,
    selectedSlot,
    battlefield,
    hand,
    canPlaceFollower,
    startDragFromHand,
    startDragFromBattlefield,
    placeOnBattlefield,
    returnToHand,
    selectFollower,
    equipItem,
    clearDrag,
  }
})
```

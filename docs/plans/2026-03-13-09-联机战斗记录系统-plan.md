# 联机战斗记录系统 实现计划

> **给 AI 助手：** 必需子技能：使用 executing-plans 或 subagent-driven-development 实现此计划。

**目标：** 实现联机战斗记录系统，招募阶段使用指令集（Command Sourcing），战斗阶段使用事件流（Event Sourcing），状态同步采用版本号 + Delta 机制。

**架构：** 分层架构 - Redis 层（状态存储）→ 系统层（Command/Battle 管理）→ 同步层（Delta 广播）。招募操作记录为指令，战斗过程记录为事件流，状态版本号确保同步可靠性。

**技术栈：** Bun Redis + RedisJSON, msgpackr, jsondiffpatch, TypeScript, Zod

// 基础游戏状态（带版本号）
export interface BaseGameState {
  id: string
  mode: 'campaign' | 'multiplayer'
  version: number            // 状态版本号，每次变更 +1
  currentRound: number
  phase: GamePhase
  subPhase?: RecruitSubPhase | CombatSubPhase
  turnTimer: number
  createdAt: number
  updatedAt: number
  players: Record<string, PlayerState>
}

// Zod schema for BaseGameState
export const baseGameStateSchema = z.object({
  id: z.string(),
  mode: z.enum(['campaign', 'multiplayer']),
  version: z.number().int().min(0),
  currentRound: z.number().int().min(1),
  phase: gamePhaseSchema,
  subPhase: z.union([recruitSubPhaseSchema, combatSubPhaseSchema]).optional(),
  turnTimer: z.number().int().min(0),
  createdAt: z.number(),
  updatedAt: z.number(),
  players: z.record(playerSchema),
})

// 单机闯关游戏状态
export interface CampaignGameState extends BaseGameState {
  mode: 'campaign'
  userId: string
}

export const campaignGameStateSchema = baseGameStateSchema.extend({
  mode: z.literal('campaign'),
  userId: z.string(),
})

// 联机游戏状态
export interface MultiplayerGameState extends BaseGameState {
  mode: 'multiplayer'
  roomId: string
  maxPlayers: number
  commandLogKey: string      // 指令日志 Key
  roundBattles: RoundBattle[]  // 每回合战斗记录
}

export const multiplayerGameStateSchema = baseGameStateSchema.extend({
  mode: z.literal('multiplayer'),
  roomId: z.string(),
  maxPlayers: z.number().int().min(2).max(10),
  commandLogKey: z.string(),
  roundBattles: z.array(z.any()), // 后续用 roundBattleSchema 替换
})

// 游戏状态联合类型
export type GameState = CampaignGameState | MultiplayerGameState

export const gameStateSchema = z.discriminatedUnion('mode', [
  campaignGameStateSchema,
  multiplayerGameStateSchema,
])

import { z } from 'zod'

// ===== 指令类型 =====

export type CommandType =
  | 'BUY'         // 购买卡牌
  | 'SELL'        // 卖出随从
  | 'REFRESH'     // 刷新商店
  | 'LOCK'        // 锁定/解锁商店
  | 'UPGRADE'     // 升级商店
  | 'MOVE'        // 移动随从位置
  | 'EQUIP'       // 装备道具
  | 'USE_SPELL'   // 使用咒术
  | 'CONFIRM'     // 确认回合

// ===== 指令载荷 =====

export interface BuyCommandPayload {
  shopSlotIndex: number
  cardId: string
  cost: number
}

export interface SellCommandPayload {
  followerInstanceId: string
  sellPrice: number
}

export interface RefreshCommandPayload {
  cost: number
}

export interface LockCommandPayload {
  locked: boolean
}

export interface UpgradeCommandPayload {
  cost: number
}

export interface MoveCommandPayload {
  followerInstanceId: string
  fromPosition: number
  toPosition: number
}

export interface EquipCommandPayload {
  followerInstanceId: string
  equipmentId: string
  cost: number
}

export interface UseSpellCommandPayload {
  spellId: string
  targetId?: string
}

export interface ConfirmCommandPayload {
  // 无额外数据
}

export type CommandPayload =
  | BuyCommandPayload
  | SellCommandPayload
  | RefreshCommandPayload
  | LockCommandPayload
  | UpgradeCommandPayload
  | MoveCommandPayload
  | EquipCommandPayload
  | UseSpellCommandPayload
  | ConfirmCommandPayload

// ===== 指令结果 =====

export interface CommandResult {
  success: boolean
  error?: string
  newGold?: number
  newState?: unknown
}

// ===== 指令定义 =====

export interface Command<T extends CommandPayload = CommandPayload> {
  id: string
  playerId: string
  type: CommandType
  timestamp: number
  payload: T
  result?: CommandResult
}

  followerInstanceId: z.string(),

  cost: z.number().int().min(0),

export const equipCommandPayloadSchema = z.object({
  followerInstanceId: z.string(),
  equipmentId: z.string(),
  cost: z.number().int().min(0),
})

export const useSpellCommandPayloadSchema = z.object({
  spellId: z.string(),
  targetId: z.string().optional(),
})

export const confirmCommandPayloadSchema = z.object({})

export const commandResultSchema = z.object({
  success: z.boolean(),
  error: z.string().optional(),
  newGold: z.number().int().optional(),
  newState: z.unknown().optional(),
})

// 基础指令 schema

  id: z.string(),
  playerId: z.string(),
  type: commandTypeSchema,
  timestamp: z.number(),
  result: commandResultSchema.optional(),

export const equipCommandSchema = baseCommandSchema.extend({

// packages/shared/src/types/battle.ts
import { z } from 'zod'

// ===== 战斗事件类型 =====

export type BattleEventType =
  | 'dice_roll'           // 拼点
  | 'first_attacker'      // 先攻决定
  | 'follower_spawn'      // 随从入场
  | 'follower_move'       // 随从移动
  | 'follower_attack'     // 随从攻击
  | 'follower_take_damage' // 随从受伤
  | 'follower_die'        // 随从死亡
  | 'follower_retreat'    // 随从撤退
  | 'keyword_trigger'     // 词条触发
  | 'status_effect_add'   // 添加状态效果
  | 'status_effect_remove' // 移除状态效果
  | 'battle_end'          // 战斗结束

// ===== 动画指令 =====

export interface AnimationInstruction {
  type: 'attack' | 'move' | 'damage' | 'heal' | 'death' | 'summon'
  fromPosition?: { x: number; y: number }
  toPosition?: { x: number; y: number }
  duration: number  // 毫秒
  delay?: number    // 延迟播放
}

export const animationInstructionSchema = z.object({
  type: z.enum(['attack', 'move', 'damage', 'heal', 'death', 'summon']),
  fromPosition: z.object({ x: z.number(), y: z.number() }).optional(),
  toPosition: z.object({ x: z.number(), y: z.number() }).optional(),
  duration: z.number().int().min(0),
  delay: z.number().int().min(0).optional(),
})

// ===== 事件载荷 =====

export interface DiceRollPayload {
  playerId: string
  diceValue: number
  modifier: number
  finalValue: number
}

export interface FirstAttackerPayload {
  playerId: string
  diceResults: {
    playerId: string
    value: number
  }[]
}

export interface FollowerSpawnPayload {
  followerInstanceId: string
  position: { x: number; y: number }
  animation: AnimationInstruction
}

export interface FollowerMovePayload {
  followerInstanceId: string
  fromPosition: { x: number; y: number }
  toPosition: { x: number; y: number }
  animation: AnimationInstruction
}

export interface FollowerAttackPayload {
  attackerId: string
  targetId: string
  damage: number
  isCritical: boolean
  triggeredKeywords?: string[]
  animation: AnimationInstruction
}

export interface FollowerTakeDamagePayload {
  followerId: string
  damage: number
  damageType: 'physical' | 'magical' | 'true'
  remainingHealth: number
  animation: AnimationInstruction
}

export interface FollowerDiePayload {
  followerId: string
  killerId?: string
  animation: AnimationInstruction
}

export interface FollowerRetreatPayload {
  followerId: string
  animation: AnimationInstruction
}

export interface KeywordTriggerPayload {
  keywordId: string
  sourceId: string
  targetIds: string[]
  effect: string
  animation?: AnimationInstruction
}

export interface StatusEffectPayload {
  followerId: string
  effectType: string
  value: number
  duration?: number
  animation?: AnimationInstruction
}

export interface BattleEndPayload {
  winnerId: string
  loserId: string
  damageDealt: number
  totalRounds: number
}

// ===== 战斗事件 =====

export interface BattleEvent {
  timestamp: number
  type: BattleEventType
  payload: unknown  // 根据类型对应不同 payload
}

// ===== Zod Schemas =====

export const battleEventTypeSchema = z.enum([
  'dice_roll', 'first_attacker', 'follower_spawn', 'follower_move',
  'follower_attack', 'follower_take_damage', 'follower_die',
  'follower_retreat', 'keyword_trigger', 'status_effect_add',
  'status_effect_remove', 'battle_end',
])

const baseEventSchema = z.object({
  timestamp: z.number(),
  type: battleEventTypeSchema,
})

export const diceRollEventSchema = baseEventSchema.extend({
  type: z.literal('dice_roll'),
  payload: z.object({
    playerId: z.string(),
    diceValue: z.number().int().min(1).max(6),
    modifier: z.number().int(),
    finalValue: z.number().int(),
  }),
})

export const firstAttackerEventSchema = baseEventSchema.extend({
  type: z.literal('first_attacker'),
  payload: z.object({
    playerId: z.string(),
    diceResults: z.array(z.object({
      playerId: z.string(),
      value: z.number(),
    })),
  }),
})

export const followerAttackEventSchema = baseEventSchema.extend({
  type: z.literal('follower_attack'),
  payload: z.object({
    attackerId: z.string(),
    targetId: z.string(),
    damage: z.number().int().min(0),
    isCritical: z.boolean(),
    triggeredKeywords: z.array(z.string()).optional(),
    animation: animationInstructionSchema,
  }),
})

// 联合事件 schema
export const battleEventSchema = z.discriminatedUnion('type', [
  diceRollEventSchema,
  firstAttackerEventSchema,

export interface BattleSnapshot {
  battleId: string

  round: z.number().int().min(1),

import type { Command, CommandResult } from '@game/shared'

export class CommandExecutor {
  constructor(private redis: any) {}

  async execute(gameId: string, command: Command): Promise<CommandResult> {
    // 根据指令类型分发
    switch (command.type) {
      case 'BUY':
        return this.executeBuy(gameId, command)
      case 'SELL':
        return this.executeSell(gameId, command)
      case 'REFRESH':
        return this.executeRefresh(gameId, command)
      case 'LOCK':
        return this.executeLock(gameId, command)
      case 'UPGRADE':
        return this.executeUpgrade(gameId, command)
      case 'MOVE':
        return this.executeMove(gameId, command)
      case 'EQUIP':
        return this.executeEquip(gameId, command)
      case 'USE_SPELL':
        return this.executeUseSpell(gameId, command)
      case 'CONFIRM':
        return this.executeConfirm(gameId, command)
      default:
        return { success: false, error: `Unknown command type: ${(command as any).type}` }
    }
  }

  private async executeBuy(gameId: string, command: any): Promise<CommandResult> {
    const { playerId, payload } = command
    const { shopSlotIndex, cost } = payload

    // 获取当前状态
    const gold = await this.redis.json.get(`game:${gameId}`, `.players.${playerId}.gold`)
    
    // 验证金币
    if (gold < cost) {
      return { success: false, error: `Insufficient gold: have ${gold}, need ${cost}` }
    }

    // 获取商店槽位
    const slot = await this.redis.json.get(
      `game:${gameId}`,
      `.players.${playerId}.shop[${shopSlotIndex}]`
    )
    
    if (!slot) {
      return { success: false, error: 'Shop slot is empty' }
    }

    // 执行购买
    const newGold = await this.redis.json.numIncrBy(
      `game:${gameId}`,
      `.players.${playerId}.gold`,
      -cost
    )

    // 清空商店槽位
    await this.redis.json.set(
      `game:${gameId}`,
      `.players.${playerId}.shop[${shopSlotIndex}]`,
      null
    )

    // 增加版本号
    await this.redis.json.numIncrBy(`game:${gameId}`, '.version', 1)

    // 记录指令日志
    await this.appendCommandLog(gameId, command)

// packages/server/src/systems/BattleRecorder.ts
import type { BattleEvent, BattleSnapshot } from '@game/shared'

export class BattleRecorder {
  constructor(private redis: any) {}

  // ===== 快照操作 =====

  async createSnapshot(battleId: string, snapshot: BattleSnapshot): Promise<void> {
    await this.redis.json.set(
      `battle:snapshot:${battleId}`,
      '.',
      snapshot
    )
  }

  async getSnapshot(battleId: string): Promise<BattleSnapshot | null> {
    const result = await this.redis.json.get(`battle:snapshot:${battleId}`)
    return result as BattleSnapshot | null
  }

  async updateSnapshot(battleId: string, path: string, value: unknown): Promise<void> {
    await this.redis.json.set(
      `battle:snapshot:${battleId}`,
      path,
      value
    )
  }

  async deleteSnapshot(battleId: string): Promise<void> {
    await this.redis.del(`battle:snapshot:${battleId}`)
  }

  // ===== 事件流操作 =====

  async recordEvent(battleId: string, event: BattleEvent): Promise<void> {
    await this.redis.lpush(
      `battle:events:${battleId}`,
      JSON.stringify(event)
    )
  }

  async recordEvents(battleId: string, events: BattleEvent[]): Promise<void> {
    const pipeline = this.redis.pipeline()
    for (const event of events) {
      pipeline.lpush(`battle:events:${battleId}`, JSON.stringify(event))
    }
    await pipeline.exec()
  }

  async getEvents(battleId: string): Promise<BattleEvent[]> {
    const results = await this.redis.lrange(`battle:events:${battleId}`, 0, -1)
    // 注意：lpush 导致顺序反转，需要反转回来
    return results
      .reverse()
      .map((r: string) => JSON.parse(r))
  }

  async getEventsInRange(
    battleId: string,
    start: number,
    end: number
  ): Promise<BattleEvent[]> {
    const results = await this.redis.lrange(
      `battle:events:${battleId}`,
      start,
      end
    )
    return results
      .reverse()
      .map((r: string) => JSON.parse(r))
  }

  async getEventCount(battleId: string): Promise<number> {
    return this.redis.llen(`battle:events:${battleId}`)
  }

  async getLastEvent(battleId: string): Promise<BattleEvent | null> {
    const result = await this.redis.lindex(`battle:events:${battleId}`, 0)
    return result ? JSON.parse(result) : null
  }

  // ===== 生命周期管理 =====

  async clearBattle(battleId: string): Promise<void> {
    await this.redis.del(`battle:snapshot:${battleId}`)
    await this.redis.del(`battle:events:${battleId}`)
  }

  async setBattleExpiry(battleId: string, ttlSeconds: number): Promise<void> {
    await this.redis.expire(`battle:snapshot:${battleId}`, ttlSeconds)
    await this.redis.expire(`battle:events:${battleId}`, ttlSeconds)
  }
}

```typescript
// packages/server/src/systems/DeltaSyncManager.ts
import { diff, Delta } from 'jsondiffpatch'
import { pack, unpack } from 'msgpackr'

export interface VersionedDelta {
  fromVersion: number
  toVersion: number
  changes: Delta
  timestamp: number
}

export class DeltaSyncManager {
  constructor(private redis: any) {}

  async recordDelta(
    gameId: string,
    oldState: any,
    newState: any
  ): Promise<VersionedDelta> {
    const changes = diff(oldState, newState)
    
    const delta: VersionedDelta = {
      fromVersion: oldState.version,
      toVersion: newState.version,
      changes: changes || {},
      timestamp: Date.now(),
    }

    // 存储到 Redis 列表（按版本号索引）
    await this.redis.lpush(
      `game:deltas:${gameId}`,
      JSON.stringify(delta)
    )

    return delta
  }

  async getDeltasSince(gameId: string, fromVersion: number): Promise<VersionedDelta[]> {
    // 获取所有 delta
    const results = await this.redis.lrange(`game:deltas:${gameId}`, 0, -1)
    
    const allDeltas: VersionedDelta[] = results
      .reverse()
      .map((r: string) => JSON.parse(r))
    
    // 过滤出需要的 delta
    return allDeltas.filter(d => d.fromVersion >= fromVersion)
  }

  async getLatestVersion(gameId: string): Promise<number> {
    const result = await this.redis.lindex(`game:deltas:${gameId}`, 0)
    if (!result) return 0
    
    const delta: VersionedDelta = JSON.parse(result)
    return delta.toVersion
  }

  needsFullSync(gameId: string, clientVersion: number, serverVersion: number): boolean {
    // 版本 0 表示新客户端，需要全量同步
    if (clientVersion === 0) return true
    
    // 版本相同，不需要同步
    if (clientVersion === serverVersion) return false
    
    // 客户端版本比服务端新，异常情况，需要全量同步
    if (clientVersion > serverVersion) return true
    
    // 检查是否有缺失的版本
    // 这里简化处理：如果差距太大，要求全量同步
    const gap = serverVersion - clientVersion
    if (gap > 100) return true // 防止 delta 列表过长
    
    return false
  }

  serializeDelta(delta: VersionedDelta): Buffer {
    return pack(delta)
  }

  deserializeDelta(data: Buffer): VersionedDelta {
    return unpack(data)
  }

  async clearDeltas(gameId: string): Promise<void> {
    await this.redis.del(`game:deltas:${gameId}`)
  }

  async getDeltaCount(gameId: string): Promise<number> {
    return this.redis.llen(`game:deltas:${gameId}`)
  }
}
```

import type { GameState, MultiplayerGameState, CampaignGameState } from '@game/shared'

// 新增方法

async createGame(
  gameId: string,
  mode: 'campaign' | 'multiplayer',
  options?: {
    roomId?: string
    maxPlayers?: number
    userId?: string
  }
): Promise<GameState> {
  const baseState = {
    id: gameId,
    mode,
    version: 0,
    currentRound: 1,
    phase: 'lobby' as const,
    turnTimer: 0,
    createdAt: Date.now(),
    updatedAt: Date.now(),
    players: {},
  }

  let state: GameState

  if (mode === 'multiplayer') {
    state = {
      ...baseState,
      mode: 'multiplayer',
      roomId: options?.roomId || '',
      maxPlayers: options?.maxPlayers || 8,
      commandLogKey: `game:${gameId}:commands`,
      roundBattles: [],
    } as MultiplayerGameState
  } else {
    state = {
      ...baseState,
      mode: 'campaign',
      userId: options?.userId || '',
    } as CampaignGameState
  }

  await this.redis.json.set(`game:${gameId}`, '.', state)
  return state
}

async getFullState(gameId: string): Promise<GameState | null> {
  const result = await this.redis.json.get(`game:${gameId}`)
  return result as GameState | null
}

async getVersion(gameId: string): Promise<number> {
  const version = await this.redis.json.get(`game:${gameId}`, '.version')
  return (version as number) || 0
}

async updateWithVersion(
  gameId: string,
  updater: (state: GameState) => GameState
): Promise<GameState> {
  // 获取当前状态
  const currentState = await this.getFullState(gameId)
  if (!currentState) {
    throw new Error(`Game not found: ${gameId}`)
  }

  // 应用更新并增加版本号
  const newState = {
    ...updater(currentState),
    version: currentState.version + 1,
    updatedAt: Date.now(),
  }

  // 保存新状态
  await this.redis.json.set(`game:${gameId}`, '.', newState)

  return newState
}

async incrementVersion(gameId: string): Promise<number> {
  const newVersion = await this.redis.json.numIncrBy(
    `game:${gameId}`,
    '.version',
    1
  )
  await this.redis.json.set(`game:${gameId}`, '.updatedAt', Date.now())
  return newVersion as number
}

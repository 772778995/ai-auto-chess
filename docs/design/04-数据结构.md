```typescript
interface Keyword {

  effect: EffectFunction
  value?: number

  }

/**
 * 关键词类型

type KeywordType =
  // 触发时机
  | 'Battlecry'      // 入场
  | 'Growth'         // 成长
  | 'Prepare'        // 准备

  | 'OnDamaged'      // 受击

  | 'Taunt'          // 嘲讽
  | 'Windfury'       // 疯狂
  | 'Counter'        // 反击

  // 特殊
  | 'Attach'         // 附着
  | 'Aura'           // 光环

 * 效果函数类型

 * 触发条件
```

## 战斗系统伪代码

```typescript
/**
 * 执行战斗
 */
function executeBattle(battlefield: Battlefield) {
  // 1. 触发成长效果
  triggerGrowthEffects(battlefield)

  // 2. 拼点决定先后攻
  const diceResult = performDiceRoll(battlefield)
  battlefield.diceRollResult = diceResult

  let attacker = diceResult.firstAttacker === 'A' ? 'A' : 'B'
  let defender = attacker === 'A' ? 'B' : 'A'

  // 3. 触发先手效果
  triggerFirstStrikeEffects(battlefield, attacker)

  // 4. 战斗循环
  while (hasAliveFollowers(battlefield, attacker) && hasAliveFollowers(battlefield, defender)) {
    const attackerFollower = getNextAttacker(battlefield, attacker)
    if (!attackerFollower) break

    const defenderFollower = selectTarget(attackerFollower, battlefield, defender)
    if (!defenderFollower) break

    // 执行攻击
    performAttack(attackerFollower, defenderFollower)

    // 检查击杀
    if (defenderFollower.currentHealth <= 0) {
      triggerOnKillEffects(attackerFollower)
      triggerDeathrattleEffects(defenderFollower, battlefield)
      removeFollower(battlefield, defenderFollower)
    }

    // 检查攻击者死亡（反击）
    if (attackerFollower.currentHealth <= 0) {
      triggerDeathrattleEffects(attackerFollower, battlefield)
      removeFollower(battlefield, attackerFollower)
    }

    // 交换攻防
    [attacker, defender] = [defender, attacker]
  }

  // 5. 计算伤害
  const result = calculateBattleResult(battlefield)
  applyDamageToHero(result)
}

/**
 * 获取下一个攻击者（横屏版）
 */
function getNextAttacker(battlefield: Battlefield, player: 'A' | 'B'): Follower | null {
  const state = player === 'A' ? battlefield.playerA : battlefield.playerB
  const side = state.side

  // 获取所有存活的随从，按位置排序
  const aliveFollowers = Array.from(state.followers.values())
    .filter(f => f.currentHealth > 0 && !f.hasAttacked)
    .sort((a, b) => getPositionNumber(a) - getPositionNumber(b))

  // 横屏攻击顺序
  // 我方(left): 从左到右 (1→2→3→4→5→6)
  // 敌方(right): 同样顺序，但视觉上镜像
  return aliveFollowers[0] || null
}

/**
 * 获取位置编号
 */
function getPositionNumber(follower: Follower): number {
  const { row, col } = follower.position
  const rowOffset = row === 'front' ? 0 : row === 'middle' ? 2 : 4
  return rowOffset + col
}

/**
 * 选择攻击目标
 */
function selectTarget(
  attacker: Follower,
  battlefield: Battlefield,
  defenderPlayer: 'A' | 'B'
): Follower | null {
  const state = defenderPlayer === 'A' ? battlefield.playerA : battlefield.playerB

  // 获取所有存活的敌方随从
  const aliveFollowers = Array.from(state.followers.values())
    .filter(f => f.currentHealth > 0)

  if (aliveFollowers.length === 0) return null

  // 按排分组
  const frontRow = aliveFollowers.filter(f => f.position.row === 'front')
  const middleRow = aliveFollowers.filter(f => f.position.row === 'middle')
  const backRow = aliveFollowers.filter(f => f.position.row === 'back')

  // 选择目标排（前排优先）
  let targetRow = frontRow.length > 0 ? frontRow :
                  middleRow.length > 0 ? middleRow : backRow

  // 检查嘲讽
  const tauntFollowers = targetRow.filter(f => f.keywords.some(k => k.type === 'Taunt'))
  if (tauntFollowers.length > 0) {
    targetRow = tauntFollowers
  }

  // 随机选择
  return targetRow[Math.floor(Math.random() * targetRow.length)]
}

/**
 * 拼点
 */
function performDiceRoll(battlefield: Battlefield): DiceRollResult {
  let rollA = Math.floor(Math.random() * 6) + 1
  let rollB = Math.floor(Math.random() * 6) + 1

  // 应用拼点加成
  rollA += getDiceRollBonus(battlefield.playerA)
  rollB += getDiceRollBonus(battlefield.playerB)

  // 如果平局，重新掷骰子（限制最大次数）
  let attempts = 0
  const maxAttempts = 10
  while (rollA === rollB && attempts < maxAttempts) {
    rollA = Math.floor(Math.random() * 6) + 1 + getDiceRollBonus(battlefield.playerA)
    rollB = Math.floor(Math.random() * 6) + 1 + getDiceRollBonus(battlefield.playerB)
    attempts++
  }

  return {
    playerA: rollA,
    playerB: rollB,
    firstAttacker: rollA >= rollB ? 'A' : 'B'
  }
}

function getDiceRollBonus(state: PlayerState): number {
  // 计算拼点加成（来自随从、装备、英雄技能等）
  let bonus = 0
  for (const follower of state.followers.values()) {
    const diceKeywords = follower.keywords.filter(k => k.type === 'DiceRoll')
    for (const keyword of diceKeywords) {
      bonus += keyword.value || 0
    }
  }
  return bonus
}
```

```typescript
  { phase: 'battle', order: 8, effect: 'Deathrattle', trigger: 'onDeath' },

  // 结算阶段
  { phase: 'settlement', order: 1, effect: 'Prepare', trigger: 'afterBattle' },
]

/**
 * 触发效果
 */
function triggerEffect(
  effectType: string,
  source: Follower,
  battlefield: Battlefield,
  params?: any
) {
  // 获取所有具有该效果的随从
  const followersWithEffect = getAllFollowers(battlefield)
    .filter(f => f.keywords.some(k => k.type === effectType))

  // 按优先级排序
  followersWithEffect.sort((a, b) => getEffectPriority(a, effectType) - getEffectPriority(b, effectType))

  // 依次触发
  for (const follower of followersWithEffect) {
    const keyword = follower.keywords.find(k => k.type === effectType)
    if (keyword && keyword.effect) {
      keyword.effect(follower, params?.target, battlefield)
    }
  }
}
```

```typescript
/**
 * 增伤系统
 */
function calculateDamage(attacker: Follower, defender: Follower): number {
  const baseDamage = attacker.currentAttack

  // 计算增伤
  const damageBoost = attacker.buffs
    .filter(b => b.type === 'damageBoost')
    .reduce((sum, b) => sum + b.value, 0)

  return baseDamage + damageBoost
}

/**
 * 三连系统
 */
function checkTriple(hand: Follower[]): { golden: Follower; reward: boolean } | null {
  const groups = new Map<string, Follower[]>()

  // 按名称分组
  for (const follower of hand) {
    if (!follower.isGolden) {
      const list = groups.get(follower.name) || []
      list.push(follower)
      groups.set(follower.name, list)
    }
  }

  // 检查是否有3个相同
  for (const [name, followers] of groups) {
    if (followers.length >= 3) {
      const golden = createGoldenFollower(followers.slice(0, 3))
      return { golden, reward: true }
    }
  }

  return null
}

/**
 * 创建金色随从
 */
function createGoldenFollower(followers: Follower[]): Follower {
  const base = followers[0]

  // 合并所有增益
  const allBuffs = followers.flatMap(f => f.buffs)
  const allEquipment = followers.flatMap(f => f.equipmentSlots)

  return {
    ...base,
    id: generateId(),
    isGolden: true,
    currentAttack: followers.reduce((sum, f) => sum + f.currentAttack, 0),
    currentHealth: followers.reduce((sum, f) => sum + f.currentHealth, 0),
    buffs: allBuffs,
    equipmentSlots: allEquipment.slice(0, 2),  // 最多保留2个装备
  }
}
```

## 光环系统

```typescript
/**
 * 光环系统
 */
function applyAuras(battlefield: Battlefield) {
  const allFollowers = getAllFollowers(battlefield)

  // 清除旧的光环效果
  for (const follower of allFollowers) {
    follower.buffs = follower.buffs.filter(b => b.source !== 'aura')
  }

  // 应用新的光环效果
  for (const follower of allFollowers) {
    const auraKeywords = follower.keywords.filter(k => k.type === 'Aura')

    for (const aura of auraKeywords) {
      const targets = getAuraTargets(follower, aura, battlefield)

      for (const target of targets) {
        applyAuraBuff(target, aura, follower.id)
      }
    }
  }
}

/**
 * 获取光环目标
 */
function getAuraTargets(
  source: Follower,
  aura: Keyword,
  battlefield: Battlefield
): Follower[] {
  const allFollowers = getAllFollowersForPlayer(battlefield, source.position.side)

  switch (aura.target?.type) {
    case 'position':
      // 针对位置（同列/同排）
      if (aura.target.value === 'column') {
        return allFollowers.filter(f => f.position.col === source.position.col)
      } else if (aura.target.value === 'row') {
        return allFollowers.filter(f => f.position.row === source.position.row)
      }
      return []

    case 'all':
      return allFollowers

    default:
      return allFollowers
  }
}

/**
 * 应用光环增益
 */
function applyAuraBuff(target: Follower, aura: Keyword, sourceId: string) {
  // 检查同名光环不叠加
  const existingAura = target.buffs.find(b =>
    b.source === 'aura' && b.sourceId === sourceId
  )

  if (existingAura) return

  // 添加光环增益
  target.buffs.push({
    id: generateId(),
    type: aura.value > 0 ? 'attack' : 'health',
    value: aura.value || 0,
    source: 'aura',
    sourceId: sourceId,
    permanent: false,
    duration: 0,
  })
}
```

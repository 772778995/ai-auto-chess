# 游戏状态管理 实现计划

> **给 AI 助手：** 必需子技能：使用 executing-plans 或 subagent-driven-development 实现此计划。

**目标：** 使用 RedisJSON 实现游戏状态管理系统，支持单机闯关和联机对战模式。

**架构：** 分层架构 - Redis 层（JSON 操作封装）→ 系统层（状态管理器）→ 服务层（业务逻辑）。状态机驱动游戏阶段转换。

**技术栈：** Bun Redis + RedisJSON, msgpackr, jsondiffpatch, TypeScript

```typescript
// packages/server/src/redis/client.ts
import { createClient, RedisClientType } from 'redis'

let client: RedisClientType | null = null

export function createRedisClient() {
  if (client) return client

  const redisUrl = process.env.REDIS_URL || 'redis://localhost:6379'
  client = createClient({ url: redisUrl })

  return {
    async connect() {
      await client!.connect()
    },

    async disconnect() {
      await client!.disconnect()
      client = null
    },

    isConnected() {
      return client?.isReady ?? false
    },

    async del(key: string) {
      return client!.del(key)
    },

    json: {
      async set(key: string, path: string, value: unknown) {
        return client!.json.set(key, path, value)
      },

      async get(key: string, path?: string) {
        return client!.json.get(key, { path: path || '.' })
      },

      async numIncrBy(key: string, path: string, increment: number) {
        return client!.json.numIncrBy(key, path, increment)
      },

      async arrAppend(key: string, path: string, ...values: unknown[]) {
        return client!.json.arrAppend(key, path, ...values)
      },
    },
  }
}
```

```typescript
// packages/server/src/systems/StateMachine.ts
export type GamePhase = 
  | 'lobby'
  | 'hero_select'
  | 'recruit'
  | 'combat'
  | 'settlement'
  | 'ended'

export type RecruitSubPhase = 'shopping' | 'positioning' | 'ready'
export type CombatSubPhase = 'dice_roll' | 'fighting' | 'result'

const STATE_TRANSITIONS: Record<GamePhase, GamePhase[]> = {
  'lobby': ['hero_select'],
  'hero_select': ['recruit'],
  'recruit': ['combat', 'ended'],
  'combat': ['settlement'],
  'settlement': ['recruit', 'ended'],
  'ended': []
}

interface StateMachineState {
  phase: GamePhase
  subPhase?: RecruitSubPhase | CombatSubPhase
  round: number
}

export function createStateMachine(initialPhase: GamePhase) {
  let state: StateMachineState = {
    phase: initialPhase,
    round: 1
  }

  return {
    getState(): GamePhase {
      return state.phase
    },

    getSubPhase(): RecruitSubPhase | CombatSubPhase | undefined {
      return state.subPhase
    },

    getRound(): number {
      return state.round
    },

    canTransitionTo(targetPhase: GamePhase): boolean {
      return STATE_TRANSITIONS[state.phase]?.includes(targetPhase) ?? false
    },

    transitionTo(targetPhase: GamePhase): void {
      if (!this.canTransitionTo(targetPhase)) {
        throw new Error(
          `Invalid transition from ${state.phase} to ${targetPhase}`
        )
      }
      state.phase = targetPhase
      state.subPhase = undefined

      if (targetPhase === 'recruit') {
        state.round++
      }
    },

    setSubPhase(subPhase: RecruitSubPhase | CombatSubPhase): void {
      state.subPhase = subPhase
    },

    getFullState(): StateMachineState {
      return { ...state }
    },

    setFullState(newState: StateMachineState): void {
      state = { ...newState }
    }
  }
}

export type StateMachine = ReturnType<typeof createStateMachine>
```

```typescript
// packages/server/src/systems/GameStateManager.ts
import type { GameState, PlayerState, GamePhase } from '@game/shared'
import type { StateMachine } from './StateMachine'
import { createStateMachine } from './StateMachine'

export class GameStateManager {
  constructor(private redis: any) {}

  async createGame(gameId: string, mode: 'campaign' | 'multiplayer'): Promise<GameState> {
    const state: GameState = {
      id: gameId,
      mode,
      currentRound: 1,
      phase: 'lobby',
      turnTimer: 0,
      createdAt: Date.now(),
      updatedAt: Date.now(),
      players: {}
    }

    await this.redis.json.set(`game:${gameId}`, '.', state)
    return state
  }

  async getGame(gameId: string): Promise<GameState | null> {
    const result = await this.redis.json.get(`game:${gameId}`)
    return result as GameState | null
  }

  async deleteGame(gameId: string): Promise<void> {
    await this.redis.del(`game:${gameId}`)
  }

  async addPlayer(gameId: string, player: Partial<PlayerState> & { id: string }): Promise<void> {
    const fullPlayer: PlayerState = {
      id: player.id,
      username: player.username || 'Unknown',
      health: player.health || 40,
      maxHealth: player.maxHealth || 40,
      gold: player.gold || 6,
      shopLevel: 1,
      shopExp: 0,
      battlefield: Array(6).fill(null),
      hand: [],
      shop: Array(6).fill(null),
      shopLocked: false,
      isReady: false,
      isAlive: true,
      ...player
    }

    await this.redis.json.set(
      `game:${gameId}`,
      `.players.${player.id}`,
      fullPlayer
    )
    await this.updateTimestamp(gameId)
  }

  async updatePlayerGold(gameId: string, playerId: string, delta: number): Promise<number> {
    const result = await this.redis.json.numIncrBy(
      `game:${gameId}`,
      `.players.${playerId}.gold`,
      delta
    )
    await this.updateTimestamp(gameId)
    return result as number
  }

  async updatePlayerHealth(gameId: string, playerId: string, delta: number): Promise<number> {
    const result = await this.redis.json.numIncrBy(
      `game:${gameId}`,
      `.players.${playerId}.health`,
      delta
    )
    await this.updateTimestamp(gameId)
    return result as number
  }

  async transitionPhase(gameId: string, targetPhase: GamePhase): Promise<void> {
    await this.redis.json.set(
      `game:${gameId}`,
      '.phase',
      targetPhase
    )
    await this.updateTimestamp(gameId)
  }

  async updateSubPhase(gameId: string, subPhase: string): Promise<void> {
    await this.redis.json.set(
      `game:${gameId}`,
      '.subPhase',
      subPhase
    )
    await this.updateTimestamp(gameId)
  }

  private async updateTimestamp(gameId: string): Promise<void> {
    await this.redis.json.set(
      `game:${gameId}`,
      '.updatedAt',
      Date.now()
    )
  }
}
```

```typescript
// packages/server/src/systems/CampaignManager.ts
import type { CampaignProgress } from '@game/shared'

export class CampaignManager {
  constructor(private redis: any) {}

  async createCampaign(userId: string): Promise<CampaignProgress> {
    const progress: CampaignProgress = {
      userId,
      currentLevel: 1,
      maxHealth: 40,
      currentHealth: 40,
      gold: 6,
      followers: [],
      equipment: [],
      spells: [],
      isVictory: false,
      isDefeated: false
    }

    await this.redis.json.set(`campaign:${userId}`, '.', progress)
    return progress
  }

  async getCampaign(userId: string): Promise<CampaignProgress | null> {
    const result = await this.redis.json.get(`campaign:${userId}`)
    return result as CampaignProgress | null
  }

  async deleteCampaign(userId: string): Promise<void> {
    await this.redis.del(`campaign:${userId}`)
  }

  async updateHealth(userId: string, delta: number): Promise<number> {
    const result = await this.redis.json.numIncrBy(
      `campaign:${userId}`,
      '.currentHealth',
      delta
    )
    
    // 检查是否死亡
    const health = result as number
    if (health <= 0) {
      await this.redis.json.set(
        `campaign:${userId}`,
        '.isDefeated',
        true
      )
    }
    
    return health
  }

  async updateGold(userId: string, delta: number): Promise<number> {
    const result = await this.redis.json.numIncrBy(
      `campaign:${userId}`,
      '.gold',
      delta
    )
    return result as number
  }

  async advanceLevel(userId: string, gameId: string): Promise<void> {
    await this.redis.json.numIncrBy(
      `campaign:${userId}`,
      '.currentLevel',
      1
    )
    await this.redis.json.set(
      `campaign:${userId}`,
      '.currentGameId',
      gameId
    )
  }

  async setVictory(userId: string): Promise<void> {
    await this.redis.json.set(
      `campaign:${userId}`,
      '.isVictory',
      true
    )
    await this.redis.json.set(
      `campaign:${userId}`,
      '.currentGameId',
      null
    )
  }

  async addFollower(userId: string, followerId: string): Promise<void> {
    // 这里需要获取完整的 follower 数据
    // 简化版本只存储 ID
    await this.redis.json.arrAppend(
      `campaign:${userId}`,
      '.followers',
      { id: followerId }
    )
  }
}
```

```typescript
// packages/server/src/systems/RoomManager.ts
import type { RoomInfo, RoomPlayer } from '@game/shared'

export class RoomManager {
  constructor(private redis: any) {}

  async createRoom(
    hostId: string,
    name: string,
    maxPlayers: number = 8
  ): Promise<RoomInfo> {
    const roomId = `room-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`
    
    const room: RoomInfo = {
      id: roomId,
      name,
      hostId,
      maxPlayers,
      players: {},
      status: 'waiting',
      createdAt: Date.now()
    }

    await this.redis.json.set(`room:${roomId}`, '.', room)
    
    // 自动添加 host 为玩家
    await this.joinRoom(roomId, hostId, `Player_${hostId.substr(0, 6)}`)
    
    return room
  }

  async getRoom(roomId: string): Promise<RoomInfo | null> {
    const result = await this.redis.json.get(`room:${roomId}`)
    return result as RoomInfo | null
  }

  async joinRoom(roomId: string, playerId: string, username: string): Promise<void> {
    const room = await this.getRoom(roomId)
    if (!room) throw new Error('Room not found')
    if (room.status !== 'waiting') throw new Error('Game already started')
    if (Object.keys(room.players).length >= room.maxPlayers) {
      throw new Error('Room is full')
    }

    const player: RoomPlayer = {
      id: playerId,
      username,
      isReady: false,
      joinedAt: Date.now()
    }

    await this.redis.json.set(`room:${roomId}`, `.players.${playerId}`, player)
    await this.redis.set(`player-room:${playerId}`, roomId)
  }

  async leaveRoom(roomId: string, playerId: string): Promise<void> {
    await this.redis.json.del(`room:${roomId}`, `.players.${playerId}`)
    await this.redis.del(`player-room:${playerId}`)
  }

  async setPlayerReady(roomId: string, playerId: string, ready: boolean): Promise<void> {
    await this.redis.json.set(
      `room:${roomId}`,
      `.players.${playerId}.isReady`,
      ready
    )
  }

  async startGame(roomId: string, gameId: string): Promise<void> {
    await this.redis.json.set(`room:${roomId}`, '.status', 'playing')
    await this.redis.json.set(`room:${roomId}`, '.currentGameId', gameId)
    await this.redis.json.set(`room:${roomId}`, '.startedAt', Date.now())
  }

  async endGame(roomId: string): Promise<void> {
    await this.redis.json.set(`room:${roomId}`, '.status', 'ended')
    await this.redis.json.set(`room:${roomId}`, '.endedAt', Date.now())
  }

  async getPlayerRoom(playerId: string): Promise<string | null> {
    return this.redis.get(`player-room:${playerId}`)
  }
}
```

```typescript
// packages/server/src/systems/DeltaRecorder.ts
import { diff, Delta } from 'jsondiffpatch'

export class DeltaRecorder {
  constructor(private redis: any) {}

  async record(gameId: string, oldState: unknown, newState: unknown): Promise<void> {
    const delta = diff(oldState, newState)
    if (!delta) return // 没有变化

    await this.redis.lpush(
      `game-delta:${gameId}`,
      JSON.stringify(delta)
    )
  }

  async getDeltas(gameId: string): Promise<Delta[]> {
    const results = await this.redis.lrange(`game-delta:${gameId}`, 0, -1)
    return results.map((r: string) => JSON.parse(r))
  }

  async getDeltaCount(gameId: string): Promise<number> {
    return this.redis.llen(`game-delta:${gameId}`)
  }

  async clearDeltas(gameId: string): Promise<void> {
    await this.redis.del(`game-delta:${gameId}`)
  }

  async getDeltasInRange(gameId: string, start: number, end: number): Promise<Delta[]> {
    const results = await this.redis.lrange(`game-delta:${gameId}`, start, end)
    return results.map((r: string) => JSON.parse(r))
  }
}
```

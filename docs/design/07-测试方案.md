### 测试用例模板

```typescript
describe('BattleSystem', () => {
  describe('🔄 拼点机制', () => {
    it('应该正确决定先后攻', () => {
      const result = performDiceRoll(mockPlayerA, mockPlayerB)
      expect(['A', 'B']).toContain(result.firstAttacker)
    })

    it('拼点加成应该正确应用', () => {
      mockPlayerA.followers = [createMockFollowerWithDiceBonus(2)]
      const result = performDiceRoll(mockPlayerA, mockPlayerB)
      expect(result.rollA).toBeGreaterThanOrEqual(3) // 1+2=3
    })
  })

  describe('🔄 横屏攻击顺序', () => {
    it('应该按位置顺序攻击', () => {
      const attackers = getAttackersInOrder(mockBattlefield, 'A')
      expect(attackers.map(f => f.position)).toEqual([
        { side: 'left', row: 'front', col: 1 },
        { side: 'left', row: 'front', col: 2 },
        // ...
      ])
    })
  })
})
```

## 1. 对手预览

```typescript
// packages/server/src/services/opponentService.ts

import { db, combatRecords } from '../db'
import { eq, and, desc } from 'drizzle-orm'

export class OpponentService {
  // 获取下回合对手
  async getNextOpponent(sessionId: string, playerId: string): Promise<string | null> {
    // 获取当前回合
    const session = await gameService.getSession(sessionId)
    const currentRound = session.currentRound

    // 获取上回合的战斗配对
    const lastCombat = await db
      .select()
      .from(combatRecords)
      .where(and(
        eq(combatRecords.sessionId, sessionId),
        eq(combatRecords.round, currentRound - 1)
      ))
      .orderBy(desc(combatRecords.createdAt))
      .limit(1)

    if (lastCombat.length === 0) return null

    const combat = lastCombat[0]
    
    // 返回对手ID
    if (combat.playerAId === playerId) return combat.playerBId
    if (combat.playerBId === playerId) return combat.playerAId

    // 上回合轮空，随机选择
    return this.getRandomOpponent(sessionId, playerId)
  }

  // 获取对手预览数据
  async getOpponentPreview(sessionId: string, playerId: string): Promise<Player | null> {
    const opponentId = await this.getNextOpponent(sessionId, playerId)
    if (!opponentId) return null

    return await gameService.getPlayerState(sessionId, opponentId)
  }

  // 随机选择对手
  private async getRandomOpponent(sessionId: string, excludePlayerId: string): Promise<string | null> {
    const alivePlayers = await gameService.getAlivePlayers(sessionId)
    const candidates = alivePlayers.filter(p => p.id !== excludePlayerId)
    
    if (candidates.length === 0) return null
    
    return candidates[Math.floor(Math.random() * candidates.length)].id
  }
}

export const opponentService = new OpponentService()
```

```vue
<template>

        _bg="dark-700"
        _rounded="md"
        _flex="~ col center"
        :class="{ 'opacity-30': !follower }"
      >
        <template v-if="follower">
          <img :src="follower.imageUrl" _w="8" _h="8" _rounded="sm" />
          <div _text="xs" _mt="1">
            <span _text="red-400">{{ follower.currentAttack }}</span>
            <span _text="gray-500">/</span>
            <span _text="green-400">{{ follower.currentHealth }}</span>
          </div>
        </template>
        <span v-else _text="gray-600">空</span>
      </div>
    </div>

    <!-- 战力估算 -->
    <div _mt="4" _text="center">
      <span _text="sm gray-400">预估战力: </span>
      <span _text="lg" :class="powerComparisonClass">
        {{ opponentPower }}
      </span>
      <span _text="sm gray-400"> vs 己方 {{ myPower }}</span>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useGameStore } from '@/stores/game'

const gameStore = useGameStore()

const opponent = computed(() => gameStore.opponentPreview)

const myPower = computed(() => gameStore.combatPower)

const opponentPower = computed(() => {
  if (!opponent.value) return 0
  return opponent.value.battlefield.reduce((total, f) => {
    if (!f) return total
    return total + f.currentAttack + f.currentHealth
  }, 0)
})

const powerComparisonClass = computed(() => {
  if (opponentPower.value > myPower.value * 1.2) return 'text-red-400'
  if (opponentPower.value < myPower.value * 0.8) return 'text-green-400'
  return 'text-yellow-400'
})
</script>
```

## 3. 排名面板 (RankingPanel.vue)

```vue
<template>
  <div _bg="dark-800" _rounded="lg" _p="4">
    <h3 _text="white" _font="bold" _mb="4">排名</h3>
    
    <div _space-y="2">
      <div
        v-for="player in sortedPlayers"
        :key="player.id"
        _flex="~" _justify="between" _items="center"
        _p="2" _rounded="md"
        :class="playerClasses(player)"
      >
        <div _flex="~" _items="center" _gap="3">
          <span 
            _w="6" _h="6"
            _rounded="full"
            _flex="~ center"
            _text="sm"
            :class="rankClass(player)"
          >
            {{ player.rank || '?' }}
          </span>
          <span _text="white">{{ player.username }}</span>
          <span v-if="player.id === currentPlayerId" _text="xs" _text="yellow-400">(你)</span>
        </div>
        
        <div _flex="~" _items="center" _gap="4">
          <div _text="sm">
            <span _text="red-400">❤️{{ player.health }}</span>
          </div>
          <div 
            _w="3" _h="3"
            _rounded="full"
            :class="player.isAlive ? 'bg-green-500' : 'bg-red-500'"
          />
        </div>
      </div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useGameStore } from '@/stores/game'

const gameStore = useGameStore()

const currentPlayerId = computed(() => gameStore.currentPlayer?.id)

const sortedPlayers = computed(() => {
  const players = [...gameStore.allPlayers]
  // 存活玩家排在前面，按生命值排序
  return players.sort((a, b) => {
    if (a.isAlive && !b.isAlive) return -1
    if (!a.isAlive && b.isAlive) return 1
    return b.health - a.health
  })
})

function playerClasses(player: Player) {
  if (player.id === currentPlayerId.value) return 'bg-blue-900/30'
  if (!player.isAlive) return 'opacity-50'
  return ''
}

function rankClass(player: Player) {
  if (!player.rank) return 'bg-gray-700 text-gray-400'
  if (player.rank === 1) return 'bg-yellow-600 text-white'
  if (player.rank === 2) return 'bg-gray-400 text-white'
  if (player.rank === 3) return 'bg-orange-600 text-white'
  return 'bg-gray-700 text-gray-400'
}
</script>
```

```typescript
// packages/web/src/composables/useCombatReplay.ts

import { ref } from 'vue'
import type { CombatResult, AttackEvent } from '@shared/types'

export function useCombatReplay() {
  const isPlaying = ref(false)
  const currentEventIndex = ref(0)
  const playbackSpeed = ref(1)

  function play(result: CombatResult): void {
    isPlaying.value = true
    currentEventIndex.value = 0

    playNextEvent(result.events)
  }

  function playNextEvent(events: AttackEvent[]): void {
    if (currentEventIndex.value >= events.length) {
      isPlaying.value = false
      return
    }

    const event = events[currentEventIndex.value]
    
    // 显示攻击动画
    showAttackAnimation(event)

    // 延迟播放下一个
    setTimeout(() => {
      currentEventIndex.value++
      playNextEvent(events)
    }, 1000 / playbackSpeed.value)
  }

  function showAttackAnimation(event: AttackEvent): void {
    // 触发动画
    // ...
  }

  function pause(): void {
    isPlaying.value = false
  }

  function setSpeed(speed: number): void {
    playbackSpeed.value = speed
  }

  return {
    isPlaying,
    currentEventIndex,
    playbackSpeed,
    play,
    pause,
    setSpeed,
  }
}
```

## 英雄数据结构

```typescript
interface Hero {
  id: string
  name: string
  type: 'passive' | 'active'
  skill?: HeroSkill
  passiveEffect?: PassiveEffect
}

interface HeroSkill {
  cost: number
  cooldown: number
  currentCooldown: number
  effect: SkillEffect
}

interface PassiveEffect {
  trigger: 'onKill' | 'onSummon' | 'onGrowth' | 'always' | 'onDiceRoll'
  bonus?: {
    attack?: number
    health?: number
    gold?: number
  }
  target?: {
    type: 'all' | 'self'
  }
}
```

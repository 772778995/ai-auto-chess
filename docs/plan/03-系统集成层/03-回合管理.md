# 03.3 回合管理

## 目标

实现招募阶段计时器、阶段切换、淘汰机制。

## 回合流程

```
┌──────────────┐    ┌──────────────┐    ┌──────────────┐
│ 招募阶段开始 │ -> │   倒计时     │ -> │  准备就绪    │
│   (60秒)    │    │  (玩家操作)  │    │  或超时      │
└──────────────┘    └──────────────┘    └──────┬───────┘
                                               │
                                               ▼
┌──────────────┐    ┌──────────────┐    ┌──────────────┐
│ 结算阶段结束 │ <- │  结算伤害    │ <- │  战斗阶段    │
│              │    │  检查淘汰    │    │  (自动进行)  │
└──────────────┘    └──────────────┘    └──────────────┘
       │
       ▼
┌──────────────┐
│ 检查游戏结束 │
│  或下一回合  │
└──────────────┘
```

## 1. 回合服务 (roundService.ts)

```typescript
// packages/server/src/services/roundService.ts

import { EventEmitter } from 'events'
import { gameService } from './gameService'
import { combatService } from './combatService'
import type { GameSession, Player } from '@shared/types'

interface RoundConfig {
  recruitTime: number      // 招募阶段时间（秒）
  combatTime: number       // 战斗阶段时间（秒）
  settlementTime: number   // 结算阶段时间（秒）
}

const DEFAULT_CONFIG: RoundConfig = {
  recruitTime: 60,
  combatTime: 30,
  settlementTime: 5,
}

export class RoundService extends EventEmitter {
  private timers: Map<string, NodeJS.Timeout> = new Map()
  private configs: Map<string, RoundConfig> = new Map()

  // 开始游戏
  async startGame(sessionId: string): Promise<void> {
    this.configs.set(sessionId, DEFAULT_CONFIG)
    
    // 进入第一回合招募阶段
    await this.startRecruitPhase(sessionId)
  }

  // 开始招募阶段
  private async startRecruitPhase(sessionId: string): Promise<void> {
    const config = this.configs.get(sessionId)!
    
    await gameService.updatePhase(sessionId, 'recruit', 'shopping')
    
    this.emit('phaseChanged', {
      sessionId,
      phase: 'recruit',
      duration: config.recruitTime,
    })

    // 启动倒计时
    let remainingTime = config.recruitTime
    
    const timer = setInterval(async () => {
      remainingTime--
      
      // 广播倒计时
      this.emit('timerUpdate', { sessionId, remainingTime })

      // 检查是否所有玩家都准备
      const allReady = await this.checkAllPlayersReady(sessionId)
      
      if (allReady || remainingTime <= 0) {
        clearInterval(timer)
        this.timers.delete(sessionId)
        await this.startCombatPhase(sessionId)
      }
    }, 1000)

    this.timers.set(sessionId, timer)
  }

  // 开始战斗阶段
  private async startCombatPhase(sessionId: string): Promise<void> {
    const config = this.configs.get(sessionId)!
    
    await gameService.updatePhase(sessionId, 'combat', 'dice_roll')
    
    this.emit('phaseChanged', {
      sessionId,
      phase: 'combat',
    })

    // 执行所有配对战斗
    await this.executeAllCombats(sessionId)

    // 进入结算阶段
    await this.startSettlementPhase(sessionId)
  }

  // 执行所有战斗
  private async executeAllCombats(sessionId: string): Promise<void> {
    const session = await gameService.getSession(sessionId)
    const players = await gameService.getAlivePlayers(sessionId)

    // 随机配对
    const pairs = this.pairPlayers(players)

    for (const [playerA, playerB] of pairs) {
      // 拼点
      const diceResult = combatService.rollDice(
        playerA.id,
        playerB.id,
        0,
        0
      )

      // 执行战斗
      const combatResult = combatService.executeCombat(
        playerA.battlefield,
        playerB.battlefield,
        diceResult.firstAttacker === playerA.id ? 'A' : 'B'
      )

      // 记录结果
      await gameService.recordCombat(sessionId, playerA.id, playerB.id, combatResult)

      // 广播战斗结果
      this.emit('combatResult', {
        sessionId,
        playerA: playerA.id,
        playerB: playerB.id,
        result: combatResult,
      })
    }

    // 处理轮空玩家（与镜像战斗）
    // ...
  }

  // 配对算法
  private pairPlayers(players: Player[]): Array<[Player, Player]> {
    // 打乱顺序
    const shuffled = [...players].sort(() => Math.random() - 0.5)
    const pairs: Array<[Player, Player]> = []

    for (let i = 0; i < shuffled.length - 1; i += 2) {
      pairs.push([shuffled[i], shuffled[i + 1]])
    }

    return pairs
  }

  // 开始结算阶段
  private async startSettlementPhase(sessionId: string): Promise<void> {
    const config = this.configs.get(sessionId)!
    
    await gameService.updatePhase(sessionId, 'settlement')
    
    // 应用伤害，检查淘汰
    await this.processSettlement(sessionId)

    // 短暂延迟后进入下一回合
    setTimeout(async () => {
      const isGameEnd = await this.checkGameEnd(sessionId)
      
      if (isGameEnd) {
        await this.endGame(sessionId)
      } else {
        // 增加回合计数
        await gameService.nextRound(sessionId)
        await this.startRecruitPhase(sessionId)
      }
    }, config.settlementTime * 1000)
  }

  // 结算处理
  private async processSettlement(sessionId: string): Promise<void> {
    const combatResults = await gameService.getCombatResults(sessionId)
    
    for (const result of combatResults) {
      // 失败者受到伤害
      const loser = result.isWin ? result.playerBId : result.playerAId
      await gameService.takeDamage(sessionId, loser, result.damageDealt)
    }

    // 检查淘汰
    const eliminated = await gameService.checkEliminations(sessionId)
    
    for (const playerId of eliminated) {
      this.emit('playerEliminated', { sessionId, playerId })
    }
  }

  // 检查是否所有玩家准备
  private async checkAllPlayersReady(sessionId: string): Promise<boolean> {
    const players = await gameService.getRoomPlayers(sessionId)
    return players.every(p => p.isReady || !p.isAlive)
  }

  // 检查游戏结束
  private async checkGameEnd(sessionId: string): Promise<boolean> {
    const alivePlayers = await gameService.getAlivePlayers(sessionId)
    return alivePlayers.length <= 1
  }

  // 结束游戏
  private async endGame(sessionId: string): Promise<void> {
    await gameService.updatePhase(sessionId, 'ended')
    
    const winner = await gameService.getWinner(sessionId)
    
    this.emit('gameEnded', { sessionId, winner })
    
    // 清理
    this.cleanup(sessionId)
  }

  // 清理
  cleanup(sessionId: string): void {
    const timer = this.timers.get(sessionId)
    if (timer) {
      clearInterval(timer)
      this.timers.delete(sessionId)
    }
    this.configs.delete(sessionId)
  }
}

export const roundService = new RoundService()
```

## 2. 客户端回合管理

```typescript
// packages/web/src/stores/round.ts

import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useWebSocket } from '@/composables/useWebSocket'
import { useGameStore } from './game'

export const useRoundStore = defineStore('round', () => {
  const gameStore = useGameStore()
  const { sendMessage, onMessage } = useWebSocket()

  // ==================== State ====================
  
  const remainingTime = ref(0)
  const isTimerRunning = ref(false)
  const isReady = ref(false)

  // ==================== Getters ====================
  
  const timeDisplay = computed(() => {
    const minutes = Math.floor(remainingTime.value / 60)
    const seconds = remainingTime.value % 60
    return `${minutes}:${seconds.toString().padStart(2, '0')}`
  })

  const isTimeRunningOut = computed(() => remainingTime.value <= 10)

  // ==================== Actions ====================
  
  // 设置准备状态
  async function setReady(ready: boolean): Promise<void> {
    isReady.value = ready
    
    sendMessage({
      type: 'player:ready',
      payload: { isReady: ready },
      timestamp: Date.now(),
    })
  }

  // 监听回合事件
  onMessage('phaseChanged', (data) => {
    gameStore.updatePhase(data.phase, data.subPhase)
    
    if (data.phase === 'recruit') {
      isReady.value = false
    }
  })

  onMessage('timerUpdate', (data) => {
    remainingTime.value = data.remainingTime
  })

  onMessage('combatResult', (data) => {
    gameStore.setCombatResult(data.result)
  })

  onMessage('playerEliminated', (data) => {
    // 显示淘汰提示
    // ...
  })

  onMessage('gameEnded', (data) => {
    // 显示游戏结束界面
    // ...
  })

  return {
    remainingTime,
    isTimerRunning,
    isReady,
    timeDisplay,
    isTimeRunningOut,
    setReady,
  }
})
```

## 4. 淘汰机制

```typescript
// packages/server/src/services/eliminationService.ts

import { db, playerGameStates, combatRecords } from '../db'
import { eq, and } from 'drizzle-orm'

export class EliminationService {
  // 检查并处理淘汰
  async processEliminations(sessionId: string): Promise<string[]> {
    const eliminated: string[] = []

    // 获取所有玩家状态
    const players = await db
      .select()
      .from(playerGameStates)
      .where(eq(playerGameStates.sessionId, sessionId))

    for (const player of players) {
      if (player.isAlive && player.health <= 0) {
        // 标记淘汰
        await db
          .update(playerGameStates)
          .set({
            isAlive: false,
            rank: await this.calculateRank(sessionId),
          })
          .where(eq(playerGameStates.id, player.id))

        eliminated.push(player.userId)
      }
    }

    return eliminated
  }

  // 计算排名
  private async calculateRank(sessionId: string): Promise<number> {
    const aliveCount = await db
      .select({ count: playerGameStates.id })
      .from(playerGameStates)
      .where(and(
        eq(playerGameStates.sessionId, sessionId),
        eq(playerGameStates.isAlive, true)
      ))

    // 排名 = 存活人数 + 1
    return aliveCount.length + 1
  }

  // 获取最终排名
  async getFinalRankings(sessionId: string): Promise<Array<{ userId: string; rank: number }>> {
    const players = await db
      .select({
        userId: playerGameStates.userId,
        rank: playerGameStates.rank,
      })
      .from(playerGameStates)
      .where(eq(playerGameStates.sessionId, sessionId))
      .orderBy(playerGameStates.rank)

    return players.map(p => ({
      userId: p.userId,
      rank: p.rank!,
    }))
  }
}

export const eliminationService = new EliminationService()
```

# 01.1 类型定义

## 目标

重构 `shared/types/game.ts`，从通用游戏类型改为卡牌自走棋专用类型。

## 现有问题

当前类型定义是通用游戏类型：
```typescript
// 当前状态 - 不匹配卡牌游戏
enum PlayerState { Idle, Walking, Running, Jumping, Attacking, Damaged, Dead }
type Player = { id, username, position, velocity, state, ... }
type GameState = { world, tick, timestamp, playerCount }
```

## 需要定义的类型

### 1. 随从类型 (Follower)

```typescript
// packages/shared/src/types/follower.ts

interface Follower {
  id: string                    // 唯一标识
  name: string                  // 名称
  description: string           // 描述
  level: 1 | 2 | 3 | 4 | 5 | 6  // 等级（决定商店出现回合）
  attack: number                // 基础攻击力
  health: number               // 基础生命值
  shield?: number              // 护盾值
  keywords: Keyword[]          // 携带的词条
  equipmentSlots: number       // 装备槽数量
  imageUrl: string             // 图片资源路径
}

// 战场上的随从实例
interface FollowerInstance extends Follower {
  instanceId: string           // 实例唯一ID
  ownerId: string             // 所属玩家ID
  position: BattlePosition    // 战场位置
  currentAttack: number       // 当前攻击力（含buff）
  currentHealth: number       // 当前生命值
  currentShield: number       // 当前护盾值
  equipment: Equipment[]      // 已装备物品
  buffs: Buff[]               // 当前buff效果
  debuffs: Debuff[]           // 当前debuff效果
}
```

### 2. 战场位置类型

```typescript
// 横屏3×2布局位置
// 己方: [0,0] [1,0] [2,0]  (前排)
//       [0,1] [1,1] [2,1]  (后排)
// 敌方: [0,0] [1,0] [2,0]  (镜像)

interface BattlePosition {
  side: 'ally' | 'enemy'     // 己方/敌方
  x: 0 | 1 | 2               // 横向位置 (0-2)
  y: 0 | 1                   // 纵向位置 (0-1)
}

// 商店位置
interface ShopPosition {
  slot: number               // 槽位编号 (0-4)
}

// 手牌位置
interface HandPosition {
  slot: number               // 槽位编号
}
```

```typescript
enum GamePhase {
```

### 4. 玩家状态类型

```typescript
interface Player {
  id: string
  username: string
  hero?: Hero                 // 选择的英雄
  health: number             // 当前生命
  maxHealth: number          // 最大生命
  gold: number               // 当前金币
  maxGold: number            // 本回合最大金币
  level: number              // 商店等级
  experience: number         // 当前经验
  experienceToNext: number   // 升级所需经验
  
  // 战场
  battlefield: (FollowerInstance | null)[]  // 6个位置

  shop: (Follower | null)[]   // 商店商品
  
  // 状态
  isReady: boolean           // 是否准备就绪
  isAlive: boolean           // 是否存活
  rank?: number              // 最终排名
  combatResult?: CombatResult // 上回合战斗结果
}
```

### 5. 词条/关键词类型

```typescript
// 触发时机
enum TriggerTiming {
  ON_ENTER = 'on_enter',         // 入场时
  ON_ATTACK = 'on_attack',       // 攻击时
  ON_HIT = 'on_hit',             // 命中时
  ON_DAMAGE = 'on_damage',       // 受到伤害时
  ON_KILL = 'on_kill',           // 击杀时
  ON_DEATH = 'on_death',         // 死亡时（遗言）
  ON_FIRST_STRIKE = 'on_first_strike', // 先手攻击时
  ON_GROWTH = 'on_growth',       // 成长时
  ON_ROUND_START = 'on_round_start', // 回合开始时
  ON_ROUND_END = 'on_round_end', // 回合结束时
  AURA = 'aura'                  // 光环效果
}

// 效果类型
enum EffectType {
  DAMAGE = 'damage',             // 造成伤害
  HEAL = 'heal',                 // 恢复生命
  SHIELD = 'shield',             // 添加护盾
  BUFF_ATTACK = 'buff_attack',   // 增加攻击
  BUFF_HEALTH = 'buff_health',   // 增加生命
  SUMMON = 'summon',             // 召唤随从
  ATTACH = 'attach',             // 附着效果
  KILL = 'kill',                 // 直接击杀
  REMOVE_KEYWORD = 'remove_keyword', // 移除词条
  ADD_KEYWORD = 'add_keyword'    // 添加词条
}

// 目标选择
enum TargetType {
  SELF = 'self',                 // 自身
  ATTACKER = 'attacker',         // 攻击者
  TARGET = 'target',             // 目标
  ALL_ALLIES = 'all_allies',     // 所有友方
  ALL_ENEMIES = 'all_enemies',   // 所有敌方
  RANDOM_ALLY = 'random_ally',   // 随机友方
  RANDOM_ENEMY = 'random_enemy', // 随机敌方
  FRONT_ROW = 'front_row',       // 前排
  BACK_ROW = 'back_row'          // 后排
}

// 关键词定义
interface Keyword {
  id: string
  name: string                  // 显示名称（如"护盾"、"嘲讽"）
  description: string           // 效果描述
  trigger?: TriggerTiming       // 触发时机
  effect: EffectDefinition      // 效果定义
  stacks?: boolean              // 是否可叠加
  visible: boolean              // 是否显示在卡牌上
}

interface EffectDefinition {
  type: EffectType
  value?: number | string       // 数值或计算公式
  target: TargetType
  condition?: Condition         // 触发条件
  duration?: number             // 持续回合数（undefined为永久）
}

interface Condition {
  type: 'health_below' | 'has_keyword' | 'position' | 'probability'
  value: any
}
```

```typescript
interface Equipment {

  level: 1 | 2 | 3 | 4 | 5 | 6

  keyword?: Keyword
}
```

### 7. 咒术类型

```typescript
interface Spell {
  id: string
  name: string
  description: string
  level: 1 | 2 | 3 | 4 | 5 | 6
  cost: number                  // 金币消耗
  effects: SpellEffect[]
  targetRequired: boolean       // 是否需要选择目标

}

interface SpellEffect {
  type: EffectType
  target: TargetType
  value: number | string
}
```

### 8. 英雄类型

```typescript
interface Hero {
  id: string
  name: string
  description: string
  skillType: 'passive' | 'active' | 'hybrid'
  passiveSkill?: PassiveSkill
  activeSkill?: ActiveSkill
  imageUrl: string
}

interface PassiveSkill {
  description: string
  effects: EffectDefinition[]
}

interface ActiveSkill {
  name: string
  description: string
  cooldown: number              // 冷却回合
  effects: SpellEffect[]
}
```

### 9. 战斗相关类型

```typescript
// 拼点结果
interface DiceRollResult {
  playerA: {
    playerId: string
    value: number              // 骰子点数
    modifier: number           // 英雄修正
    finalValue: number         // 最终值
  }
  playerB: {
    playerId: string
    value: number
    modifier: number
    finalValue: number
  }
  firstAttacker: string        // 先攻玩家ID
}

// 攻击事件
interface AttackEvent {
  attacker: FollowerInstance
  target: FollowerInstance
  damage: number
  isCritical: boolean
  triggeredKeywords: Keyword[]
}

// 战斗结果
interface CombatResult {
  opponentId: string
  isWin: boolean
  damageDealt: number         // 对敌方英雄造成的伤害
  rounds: number              // 战斗持续回合数
  mvpFollower?: string        // MVP随从ID
  events: AttackEvent[]       // 战斗事件记录
}
```

### 10. 游戏房间类型

```typescript
interface GameRoom {
  id: string
  name: string
  hostId: string
  players: Player[]
  maxPlayers: number          // 通常为8
  status: 'waiting' | 'playing' | 'ended'
  createdAt: number
  startedAt?: number
  endedAt?: number
}

interface GameSession {
  roomId: string
  currentRound: number        // 当前回合数
  phase: GamePhase
  subPhase: RecruitSubPhase | CombatSubPhase
  turnTimer: number           // 倒计时
  timestamp: number
}
```

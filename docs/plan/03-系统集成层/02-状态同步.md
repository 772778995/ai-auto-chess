# 03.2 状态同步

## 目标

实现 WebSocket 状态广播和增量更新。

## 同步策略

```
┌─────────────┐     ┌─────────────┐     ┌─────────────┐
│   服务端     │     │   序列化    │     │   客户端    │
│  状态变更   │ ->  │ msgpackr   │ ->  │  应用更新   │
└─────────────┘     └─────────────┘     └─────────────┘
       │                   │                   │
       │  1.计算增量       │  2.二进制压缩     │  3.解压应用
       │  (jsondiffpatch)  │  (高效传输)       │  (状态更新)
```

```typescript
export enum WsMessageType {
  // 连接管理
  HANDSHAKE = 'handshake',
  PING = 'ping',
  PONG = 'pong',
  
  // 房间管理
  ROOM_JOIN = 'room:join',
  ROOM_LEAVE = 'room:leave',
  ROOM_UPDATE = 'room:update',
  
  // 游戏状态
  GAME_STATE_FULL = 'game:state:full',      // 完整状态
  GAME_STATE_DELTA = 'game:state:delta',    // 增量更新
  GAME_PHASE_CHANGE = 'game:phase:change',
  
  // 玩家操作
  PLAYER_READY = 'player:ready',
  PLAYER_ACTION = 'player:action',
  PLAYER_CHAT = 'player:chat',
  
  // 战斗
  COMBAT_START = 'combat:start',
  COMBAT_RESULT = 'combat:result',
  
  // 匹配
  MATCH_FOUND = 'match:found',
  MATCH_CANCELLED = 'match:cancelled',
}

export interface WsMessage<T = unknown> {
  type: WsMessageType
  payload: T
  timestamp: number
  senderId?: string
}
```

## 2. 状态同步服务 (syncService.ts)

```typescript
// packages/server/src/services/syncService.ts

import { create } from 'jsondiffpatch'
import { pack, unpack } from 'msgpackr'
import type { GameState } from '@shared/types'

const diffPatcher = create({
  objectHash: (obj: any) => obj.id || JSON.stringify(obj),
  arrays: { detectMove: true },
})

export class SyncService {
  // 存储每个房间的上一个状态
  private lastStates: Map<string, GameState> = new Map()

  // 计算并发送增量更新
  computeDelta(roomId: string, newState: GameState): Buffer {
    const lastState = this.lastStates.get(roomId)
    
    if (!lastState) {
      // 第一次发送完整状态
      this.lastStates.set(roomId, newState)
      return pack({
        type: 'full',
        data: newState,
      })
    }

    // 计算差异
    const delta = diffPatcher.diff(lastState, newState)
    
    if (!delta) {
      // 没有变化
      return pack({ type: 'noop' })
    }

    // 更新存储的状态
    this.lastStates.set(roomId, newState)

    // 返回增量
    return pack({
      type: 'delta',
      data: delta,
    })
  }

  // 获取完整状态（用于重连）
  getFullState(roomId: string): GameState | undefined {
    return this.lastStates.get(roomId)
  }

  // 清理房间状态
  cleanupRoom(roomId: string): void {
    this.lastStates.delete(roomId)
  }
}

export const syncService = new SyncService()
```

## 3. WebSocket 处理 (gameSocket.ts)

```typescript
// packages/server/src/sockets/gameSocket.ts

import type { ServerWebSocket } from 'bun'
import { syncService } from '../services/syncService'
import { gameService } from '../services/gameService'
import { WsMessageType, type WsMessage } from '@shared/types'

interface WebSocketData {
  userId: string
  roomId?: string
}

export function handleGameSocket(ws: ServerWebSocket<WebSocketData>): void {
  ws.subscribe(`room:${ws.data.roomId}`)

  // 发送当前状态
  const fullState = syncService.getFullState(ws.data.roomId!)
  if (fullState) {
    ws.send(syncService.computeDelta(ws.data.roomId!, fullState))
  }

  ws.onmessage = (event) => {
    try {
      const message: WsMessage = JSON.parse(event.data as string)
      handleMessage(ws, message)
    } catch (error) {
      console.error('Invalid WebSocket message:', error)
    }
  }

  ws.onclose = () => {
    ws.unsubscribe(`room:${ws.data.roomId}`)
    handleDisconnect(ws)
  }
}

async function handleMessage(
  ws: ServerWebSocket<WebSocketData>,
  message: WsMessage
): Promise<void> {
  const { type, payload } = message

  switch (type) {
    case WsMessageType.PLAYER_READY:
      await handlePlayerReady(ws, payload)
      break

    case WsMessageType.PLAYER_ACTION:
      await handlePlayerAction(ws, payload)
      break

    case WsMessageType.PING:
      ws.send(JSON.stringify({ type: WsMessageType.PONG }))
      break

    default:
      console.warn('Unknown message type:', type)
  }
}

async function handlePlayerReady(
  ws: ServerWebSocket<WebSocketData>,
  payload: { isReady: boolean }
): Promise<void> {
  const { roomId, userId } = ws.data
  if (!roomId) return

  // 更新准备状态
  await roomService.setReady(roomId, userId, payload.isReady)

  // 广播更新
  broadcastToRoom(roomId, {
    type: WsMessageType.PLAYER_READY,
    payload: { userId, isReady: payload.isReady },
    timestamp: Date.now(),
  })

  // 检查是否所有玩家都准备
  const canStart = await roomService.canStartGame(roomId)
  if (canStart) {
    startGame(roomId)
  }
}

async function handlePlayerAction(
  ws: ServerWebSocket<WebSocketData>,
  payload: { action: string; data: any }
): Promise<void> {
  const { roomId, userId } = ws.data
  if (!roomId) return

  // 执行操作
  const newState = await gameService.executeAction(roomId, userId, payload)

  // 计算增量并广播
  const delta = syncService.computeDelta(roomId, newState)
  ws.publish(`room:${roomId}`, delta)
}

function broadcastToRoom(roomId: string, message: WsMessage): void {
  // 使用 Bun 的 pub/sub
  // server.publish(`room:${roomId}`, JSON.stringify(message))
}

async function handleDisconnect(ws: ServerWebSocket<WebSocketData>): Promise<void> {
  const { roomId, userId } = ws.data
  if (!roomId) return

  // 标记为离线
  await gameService.setPlayerOffline(roomId, userId)

  // 广播断开连接
  broadcastToRoom(roomId, {
    type: WsMessageType.PLAYER_LEAVE,
    payload: { userId },
    timestamp: Date.now(),
  })
}
```

## 4. 客户端同步 (useSync.ts)

```typescript
// packages/web/src/composables/useSync.ts

import { ref, watch } from 'vue'
import { create } from 'jsondiffpatch'
import { unpack } from 'msgpackr'
import { useGameStore } from '@/stores/game'
import { useWebSocket } from './useWebSocket'

const diffPatcher = create()

export function useSync() {
  const gameStore = useGameStore()
  const { onBinaryMessage } = useWebSocket()

  const isConnected = ref(false)
  const lastUpdateTime = ref(0)
  const updateCount = ref(0)

  // 处理二进制消息（增量更新）
  onBinaryMessage((data) => {
    try {
      const message = unpack(data)
      
      switch (message.type) {
        case 'full':
          // 完整状态
          gameStore.initGame(message.data.session, message.data.player)
          break

        case 'delta':
          // 增量更新
          applyDelta(message.data)
          break

        case 'noop':
          // 无变化
          break
      }

      lastUpdateTime.value = Date.now()
      updateCount.value++
    } catch (error) {
      console.error('Failed to process sync update:', error)
    }
  })

  // 应用增量更新
  function applyDelta(delta: any): void {
    const currentState = {
      session: {
        phase: gameStore.phase,
        subPhase: gameStore.subPhase,
        currentRound: gameStore.currentRound,
        turnTimer: gameStore.turnTimer,
      },
      player: gameStore.currentPlayer,
    }

    // 使用 jsondiffpatch 应用差异
    const newState = diffPatcher.patch(currentState, delta)

    // 更新状态
    gameStore.updatePhase(newState.session.phase, newState.session.subPhase)
    gameStore.currentRound = newState.session.currentRound
    gameStore.turnTimer = newState.session.turnTimer
    if (newState.player) {
      gameStore.updatePlayer(newState.player)
    }
  }

  // 请求重连（获取完整状态）
  async function requestReconnect(): Promise<void> {
    // 发送重连请求
    // ...
  }

  return {
    isConnected,
    lastUpdateTime,
    updateCount,
    requestReconnect,
  }
}
```

## 5. 断线重连机制

```typescript
// packages/web/src/composables/useReconnect.ts

import { ref, onMounted, onUnmounted } from 'vue'
import { useWebSocket } from './useWebSocket'
import { useGameStore } from '@/stores/game'

const RECONNECT_INTERVAL = 3000
const MAX_RECONNECT_ATTEMPTS = 5

export function useReconnect() {
  const { connect, disconnect, isConnected } = useWebSocket()
  const gameStore = useGameStore()

  const reconnectAttempts = ref(0)
  const isReconnecting = ref(false)
  let reconnectTimer: number | null = null

  function startReconnect() {
    if (isReconnecting.value) return
    isReconnecting.value = true

    reconnectTimer = window.setInterval(async () => {
      if (isConnected.value) {
        // 重连成功
        stopReconnect()
        return
      }

      if (reconnectAttempts.value >= MAX_RECONNECT_ATTEMPTS) {
        // 重连失败，返回大厅
        stopReconnect()
        gameStore.resetGame()
        return
      }

      reconnectAttempts.value++
      
      try {
        await connect()
      } catch (error) {
        console.error('Reconnect attempt failed:', error)
      }
    }, RECONNECT_INTERVAL)
  }

  function stopReconnect() {
    if (reconnectTimer) {
      clearInterval(reconnectTimer)
      reconnectTimer = null
    }
    isReconnecting.value = false
    reconnectAttempts.value = 0
  }

  // 监听连接断开
  watch(isConnected, (connected) => {
    if (!connected && gameStore.sessionId) {
      startReconnect()
    }
  })

  return {
    isReconnecting,
    reconnectAttempts,
    startReconnect,
    stopReconnect,
  }
}
```

## 1. 商店组件 (ShopPanel.vue)

```vue
<!-- packages/web/src/components/game/ShopPanel.vue -->
<template>
  <div _bg="dark-800" _rounded="lg" _p="4">
    <!-- 等级和升级 -->
    <div _flex="~" _justify="between" _items="center" _mb="4">
      <div _flex="~" _items="center" _gap="2">
        <span _text="white" _font="bold">等级: {{ currentLevel }}</span>
        <div _w="32" _h="2" _bg="dark-600" _rounded="full">
          <div 
            _h="full" _bg="yellow-500" _rounded="full"
            :style="{ width: `${(currentExp / expToNext) * 100}%` }"
          />
        </div>
        <span _text="gray-400" _text="sm">{{ currentExp }}/{{ expToNext }}</span>
      </div>
      
      <button
        _game-btn
        _px="3" _py="1"
        :disabled="!canUpgrade"
        @click="upgradeShop"
      >
        升级 ({{ upgradeCost }}金)
      </button>
    </div>
    
    <!-- 商店槽位 -->
    <div _flex="~" _gap="3" _mb="4">
      <ShopSlot
        v-for="(slot, index) in shopSlots"
        :key="index"
        :follower="slot"
        :index="index"
        @buy="buyFollower(index)"
        @hover="showDetails"
      />
    </div>
    
    <!-- 操作按钮 -->
    <div _flex="~" _gap="3" _justify="center">
      <button
        _game-btn
        :disabled="!canRefresh"
        @click="refreshShop"
      >
        刷新 (1金)
      </button>
      
      <button
        _game-btn
        :class="{ 'bg-yellow-600': isLocked }"
        @click="toggleLock"
      >
        {{ isLocked ? '已锁定' : '锁定' }}
      </button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import { useShopStore } from '@/stores/shop'
import { useGameStore } from '@/stores/game'
import ShopSlot from './ShopSlot.vue'

const shopStore = useShopStore()
const gameStore = useGameStore()

const currentLevel = computed(() => gameStore.currentPlayer?.level || 1)
const currentExp = computed(() => gameStore.currentPlayer?.experience || 0)
const expToNext = computed(() => gameStore.currentPlayer?.experienceToNext || 2)

const upgradeCost = computed(() => {
  // 升级花费: 等级2=4金, 等级3=6金, 等级4=8金...
  return currentLevel.value * 2
})

const canUpgrade = computed(() => {
  return (gameStore.currentPlayer?.gold || 0) >= upgradeCost.value
})

const { shopSlots, isLocked, canRefresh, refreshShop, toggleLock, buyFollower } = shopStore

function upgradeShop() {
  if (!canUpgrade.value) return
  
  // 扣除金币
  gameStore.addGold(-upgradeCost.value)
  
  // 增加经验
  gameStore.addExperience(4) // 购买4点经验
}

function showDetails(follower: Follower) {
  // 显示随从详情
}
</script>
```

## 2. 商店槽位组件 (ShopSlot.vue)

```vue
<!-- packages/web/src/components/game/ShopSlot.vue -->
<template>
  <div
    _w="20" _h="28"
    _bg="dark-700"
    _rounded="lg"
    _border="2"
    :class="borderClass"
    _cursor="pointer"
    _transition="all"
    _hover:scale="105"
    @click="handleClick"
    @mouseenter="$emit('hover', follower)"
  >
    <template v-if="follower">
      <!-- 随从图片 -->
      <div _h="14" _bg="dark-600" _rounded="t-lg" _overflow="hidden">
        <img :src="follower.imageUrl" _w="full" _h="full" _object="cover" />
      </div>
      
      <!-- 星级 -->
      <div _flex="~" _justify="center" _gap="0.5" _py="1">
        <span
          v-for="i in follower.level"
          :key="i"
          _text="yellow-400" _text="xs"
        >★</span>
      </div>
      
      <!-- 属性和价格 -->
      <div _px="2" _pb="2">
        <div _flex="~" _justify="between" _text="xs">
          <span _text="red-400">⚔️{{ follower.attack }}</span>
          <span _text="green-400">❤️{{ follower.health }}</span>
        </div>
        <div _text="center" _text="yellow-400" _font="bold" _mt="1">
          {{ cost }}金
        </div>
      </div>
    </template>
    
    <!-- 空槽位 -->
    <div v-else _h="full" _flex="~ center" _text="gray-600">
      空
    </div>
  </div>
</template>

<script setup lang="ts">
import { computed } from 'vue'
import type { Follower } from '@shared/types'

const props = defineProps<{
  follower: Follower | null
  index: number
}>()

const emit = defineEmits<{
  buy: [index: number]
  hover: [follower: Follower]
}>()

const cost = computed(() => {
  // 所有随从都是3金币
  return 3
})

const borderClass = computed(() => {
  if (!props.follower) return 'border-dark-600'
  // 根据等级显示不同颜色边框
  const colors = ['', 'border-gray-500', 'border-green-500', 'border-blue-500', 
                  'border-purple-500', 'border-orange-500', 'border-red-500']
  return colors[props.follower.level] || 'border-gray-500'
})

function handleClick() {
  if (props.follower) {
    emit('buy', props.index)
  }
}
</script>
```

  1: 3,

  3: 5,
  4: 6,
  5: 7,
  // 之后每回合都是7金币

// 利息计算 (每10金币存1金币利息，最多5金币)
export function calculateInterest(savedGold: number): number {
  return Math.min(Math.floor(savedGold / 10), 5)
}

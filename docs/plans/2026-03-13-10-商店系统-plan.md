# 商店系统实现计划

> **给 AI 助手：** 必需子技能：使用 executing-plans 或 subagent-driven-development 实现此计划。

**目标：** 实现完整的商店系统，包括刷新、购买、出售、升级功能，支持金币与经验双轨升级机制。

**架构：** 商店系统由 CardPool（卡池管理）、ShopConfig（配置常量）、ShopType（类型定义）和 ShopSystem（核心逻辑）组成，通过 GameStateManager 读写 RedisJSON 状态。

**技术栈：** TypeScript + Zod + RedisJSON + Bun Test

```typescript
 * 获取升级所需经验

  const config = SHOP_LEVEL_CONFIG[level]

 * 检查是否可以升级

  return currentExp >= required

 * 根据商店等级获取星级概率分布

 * 根据概率分布随机选择星级

  const distribution = getStarDistribution(level)
  const rand = Math.random() * 100
  let cumulative = 0
  
  for (const [star, prob] of Object.entries(distribution)) {

    if (rand <= cumulative) {
      return parseInt(star)

}

  return config?.upgradeReward ?? null
```

```typescript
import type { Follower } from '@game/shared/src/types/follower'
import type { Spell } from '@game/shared/src/data/spells'
import type { ShopSlot } from '@game/shared/src/types/shop'

// 装备类型（简化版，后续可扩展）
export interface Equipment {
  id: string
  name: string
  level: number
}

export interface DrawResult {
  type: 'follower' | 'equipment' | 'spell'
  cardId: string
}

export class CardPool {
  constructor(
    private followers: Follower[],
    private equipment: Equipment[],
    private spells: Spell[]
  ) {}

  /**
   * 从卡池抽取指定类型和星级的卡牌
   */
  draw(type: 'follower' | 'equipment', starLevel: number): DrawResult | null {
    if (type === 'follower') {
      const candidates = this.followers.filter(f => f.level === starLevel)
      if (candidates.length === 0) return null
      const card = candidates[Math.floor(Math.random() * candidates.length)]
      return { type: 'follower', cardId: card.id }
    }

    if (type === 'equipment') {
      const candidates = this.equipment.filter(e => e.level === starLevel)
      if (candidates.length === 0) return null
      const card = candidates[Math.floor(Math.random() * candidates.length)]
      return { type: 'equipment', cardId: card.id }
    }

    return null
  }

  /**
   * 根据商店等级随机抽取（考虑装备概率）
   * @param shopLevel 商店等级
   * @param equipmentProb 装备出现概率（0-1）
   * @param starLevel 指定星级，不提供则根据 shopLevel 随机
   */
  drawRandom(
    shopLevel: number,
    equipmentProb: number,
    starLevel?: number
  ): DrawResult | null {
    // 1级商店只有随从
    const canDrawEquipment = shopLevel >= 2 && this.equipment.length > 0
    const isEquipment = canDrawEquipment && Math.random() < equipmentProb

    const targetLevel = starLevel ?? this.rollStarLevel(shopLevel)

    if (isEquipment) {
      const result = this.draw('equipment', targetLevel)
      if (result) return result
    }

    // 默认抽随从
    return this.draw('follower', targetLevel)
  }

  /**
   * 获取随机咒术（用于升级奖励）
   */
  getRandomSpell(level: number): Spell | null {
    const candidates = this.spells.filter(s => s.level === level)
    if (candidates.length === 0) return null
    return candidates[Math.floor(Math.random() * candidates.length)]
  }

  /**
   * 根据商店等级滚动星级
   */
  private rollStarLevel(shopLevel: number): number {
    const distribution = this.getStarDistribution(shopLevel)
    const rand = Math.random() * 100
    let cumulative = 0

    for (const [star, prob] of Object.entries(distribution)) {
      cumulative += prob
      if (rand <= cumulative) {
        return parseInt(star)
      }
    }

    return 1
  }

  /**
   * 获取星级概率分布
   */
  private getStarDistribution(shopLevel: number): Record<number, number> {
    const distributions: Record<number, Record<number, number>> = {
      1: { 1: 100 },
      2: { 1: 60, 2: 40 },
      3: { 1: 10, 2: 30, 3: 60 },
      4: { 1: 10, 2: 20, 3: 30, 4: 40 },
      5: { 2: 11, 3: 22, 4: 37, 5: 30 },
      6: { 3: 25, 4: 35, 5: 30, 6: 10 },
    }

    return distributions[shopLevel] ?? { 1: 100 }
  }
}
```

```typescript
import type { CardPool } from './CardPool'
import type { 
  ShopSlot, 
  BuyResult, 
  SellResult, 
  RefreshResult, 
  UpgradeResult,
  PriceCalculator 
} from '@game/shared/src/types/shop'
import { 
  SHOP_CONFIG, 
  getExpRequiredForLevel, 
  canLevelUp, 
  getUpgradeReward 
} from '@game/shared/src/constants/shop'

// GameStateManager 接口（商店系统需要的部分）
interface IGameStateManager {
  getPlayerState(gameId: string, playerId: string): Promise<{
    gold: number
    shopLevel: number
    shopExp: number
    shopLocked?: boolean
    shop: (ShopSlot | null)[]
    hand: unknown[]
    battlefield: unknown[]
  } | null>
  updatePlayerGold(gameId: string, playerId: string, delta: number): Promise<number>
  updatePlayerShopExp(gameId: string, playerId: string, delta: number): Promise<number>
  updatePlayerShopLevel(gameId: string, playerId: string, newLevel: number, newExp: number): Promise<void>
  updatePlayerShop(gameId: string, playerId: string, shop: (ShopSlot | null)[]): Promise<void>
  setShopLocked(gameId: string, playerId: string, locked: boolean): Promise<void>
  addCardToHand(gameId: string, playerId: string, cardId: string, type: 'follower' | 'equipment'): Promise<void>
  removeCardFromHand(gameId: string, playerId: string, instanceId: string): Promise<boolean>
  removeCardFromBattlefield(gameId: string, playerId: string, instanceId: string): Promise<boolean>
  addSpellToHand(gameId: string, playerId: string, spellId: string): Promise<void>
}

export class ShopSystem {
  constructor(
    private cardPool: CardPool,
    private gameStateManager: IGameStateManager,
    private priceCalculator?: PriceCalculator
  ) {}

  /**
   * 刷新商店
   * 花费金币刷新商店槽位（保留锁定的槽位）
   */
  async refresh(gameId: string, playerId: string): Promise<RefreshResult> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) {
      return { success: false, cost: 0, slots: [], error: '玩家不存在' }
    }

    const refreshCost = this.priceCalculator?.refreshCost(SHOP_CONFIG.REFRESH_COST, playerState) 
      ?? SHOP_CONFIG.REFRESH_COST

    // 检查金币
    if (playerState.gold < refreshCost) {
      return { success: false, cost: refreshCost, slots: playerState.shop, error: '金币不足' }
    }

    // 扣除金币
    await this.gameStateManager.updatePlayerGold(gameId, playerId, -refreshCost)

    // 生成新槽位（保留锁定槽位）
    const newSlots = await this.generateSlots(playerState.shop, playerState.shopLevel)
    await this.gameStateManager.updatePlayerShop(gameId, playerId, newSlots)

    return { success: true, cost: refreshCost, slots: newSlots }
  }

  /**
   * 购买卡牌
   */
  async buy(gameId: string, playerId: string, slotIndex: number): Promise<BuyResult> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) {
      return { success: false, error: '玩家不存在' }
    }

    const slot = playerState.shop[slotIndex]
    if (!slot) {
      return { success: false, error: '槽位为空' }
    }

    const buyPrice = this.priceCalculator?.buyPrice(SHOP_CONFIG.BUY_BASE_PRICE, playerState, slot.type) 
      ?? SHOP_CONFIG.BUY_BASE_PRICE

    // 检查金币
    if (playerState.gold < buyPrice) {
      return { success: false, error: '金币不足' }
    }

    // 扣除金币
    await this.gameStateManager.updatePlayerGold(gameId, playerId, -buyPrice)

    // 添加卡牌到手牌
    if (slot.type === 'follower' || slot.type === 'equipment') {
      await this.gameStateManager.addCardToHand(gameId, playerId, slot.cardId, slot.type)
    }

    // 清空槽位
    const newShop = [...playerState.shop]
    newShop[slotIndex] = null
    await this.gameStateManager.updatePlayerShop(gameId, playerId, newShop)

    return { 
      success: true, 
      cardId: slot.cardId, 
      cardType: slot.type 
    }
  }

  /**
   * 出售卡牌
   */
  async sell(
    gameId: string, 
    playerId: string, 
    cardId: string, 
    from: 'hand' | 'battlefield'
  ): Promise<SellResult> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) {
      return { success: false, goldGained: 0, error: '玩家不存在' }
    }

    const sellPrice = this.priceCalculator?.sellPrice(SHOP_CONFIG.SELL_BASE_PRICE, playerState, 'follower') 
      ?? SHOP_CONFIG.SELL_BASE_PRICE

    // 移除卡牌
    let removed = false
    if (from === 'hand') {
      removed = await this.gameStateManager.removeCardFromHand(gameId, playerId, cardId)
    } else {
      removed = await this.gameStateManager.removeCardFromBattlefield(gameId, playerId, cardId)
    }

    if (!removed) {
      return { success: false, goldGained: 0, error: '卡牌不存在' }
    }

    // 增加金币
    await this.gameStateManager.updatePlayerGold(gameId, playerId, sellPrice)

    return { success: true, goldGained: sellPrice }
  }

  /**
   * 升级商店（花费金币换经验）
   */
  async upgrade(gameId: string, playerId: string, goldAmount: number): Promise<UpgradeResult> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) {
      return { success: false, expAdded: 0, expRemaining: 0, error: '玩家不存在' }
    }

    if (playerState.shopLevel >= 6) {
      return { success: false, expAdded: 0, expRemaining: 0, error: '已达到最高等级' }
    }

    if (playerState.gold < goldAmount) {
      return { success: false, expAdded: 0, expRemaining: 0, error: '金币不足' }
    }

    // 扣除金币，增加经验（1金币 = 1经验）
    await this.gameStateManager.updatePlayerGold(gameId, playerId, -goldAmount)
    const newExp = await this.gameStateManager.updatePlayerShopExp(gameId, playerId, goldAmount)

    const oldLevel = playerState.shopLevel
    let newLevel = oldLevel
    let remainingExp = newExp
    let rewards = undefined

    // 检查升级
    while (canLevelUp(newLevel, remainingExp)) {
      const expRequired = getExpRequiredForLevel(newLevel + 1)
      remainingExp -= expRequired
      newLevel++

      // 发放升级奖励
      const reward = getUpgradeReward(newLevel)
      if (reward) {
        for (let i = 0; i < reward.count; i++) {
          const spell = this.cardPool.getRandomSpell(reward.spellLevel)
          if (spell) {
            await this.gameStateManager.addSpellToHand(gameId, playerId, spell.id)
          }
        }
        rewards = reward
      }

      if (newLevel >= 6) break
    }

    // 更新等级
    if (newLevel > oldLevel) {
      await this.gameStateManager.updatePlayerShopLevel(gameId, playerId, newLevel, remainingExp)
    }

    return {
      success: true,
      oldLevel,
      newLevel,
      expAdded: goldAmount,
      expRemaining: remainingExp,
      rewards
    }
  }

  /**
   * 切换商店锁定状态
   */
  async toggleLock(gameId: string, playerId: string): Promise<boolean> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) return false

    const newLocked = !(playerState.shopLocked ?? false)
    await this.gameStateManager.setShopLocked(gameId, playerId, newLocked)
    return newLocked
  }

  /**
   * 回合开始处理（自动+经验、自动升级、自动刷新）
   */
  async onRoundStart(gameId: string, playerId: string): Promise<void> {
    const playerState = await this.gameStateManager.getPlayerState(gameId, playerId)
    if (!playerState) return

    // +2 经验
    let newExp = await this.gameStateManager.updatePlayerShopExp(
      gameId, 
      playerId, 
      SHOP_CONFIG.EXP_PER_ROUND
    )

    let currentLevel = playerState.shopLevel

    // 自动升级检查
    while (canLevelUp(currentLevel, newExp)) {
      const expRequired = getExpRequiredForLevel(currentLevel + 1)
      newExp -= expRequired
      currentLevel++

      // 发放升级奖励
      const reward = getUpgradeReward(currentLevel)
      if (reward) {
        for (let i = 0; i < reward.count; i++) {
          const spell = this.cardPool.getRandomSpell(reward.spellLevel)
          if (spell) {
            await this.gameStateManager.addSpellToHand(gameId, playerId, spell.id)
          }
        }
      }

      if (currentLevel >= 6) break
    }

    if (currentLevel > playerState.shopLevel) {
      await this.gameStateManager.updatePlayerShopLevel(gameId, playerId, currentLevel, newExp)
    }

    // 自动刷新（如果未锁定）
    if (!playerState.shopLocked) {
      await this.refresh(gameId, playerId)
    }
  }

  /**
   * 生成商店槽位（保留锁定槽位）
   */
  private async generateSlots(
    currentSlots: (ShopSlot | null)[], 
    shopLevel: number
  ): Promise<(ShopSlot | null)[]> {
    const newSlots: (ShopSlot | null)[] = [...currentSlots]

    for (let i = 0; i < SHOP_CONFIG.SLOT_COUNT; i++) {
      const currentSlot = currentSlots[i]
      
      // 保留锁定槽位
      if (currentSlot?.isLocked) {
        newSlots[i] = currentSlot
        continue
      }

      // 抽取新卡
      const card = this.cardPool.drawRandom(
        shopLevel,
        SHOP_CONFIG.EQUIPMENT_PROBABILITY
      )

      if (card) {
        newSlots[i] = {
          type: card.type as 'follower' | 'equipment' | 'spell',
          cardId: card.cardId,
          isLocked: false,
          discount: 0
        }
      } else {
        newSlots[i] = null
      }
    }

    return newSlots
  }
}
```

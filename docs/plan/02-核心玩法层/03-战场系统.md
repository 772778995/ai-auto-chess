# 02.3 战场系统

## 目标

实现横屏3×2战场布局，支持随从放置、移除、拖拽调整站位。

## 1. 战场组件 (Battlefield.vue)

```vue
<!-- packages/web/src/components/game/Battlefield.vue -->
<template>
  <div _w="full" _h="full" _flex="~ col" _gap="4" _p="4">
    <!-- 敌方战场 -->
    <div _flex="~ col" _gap="2">
      <div _text="gray-400" _text="sm" _text="center">敌方</div>
      <div _flex="~" _justify="center" _gap="2">
        <BattleSlot
          v-for="index in 6"
          :key="`enemy-${index - 1}`"
          :follower="enemyBattlefield[index - 1]"
          :is-enemy="true"
          :slot-index="index - 1"
        />
      </div>
    </div>
    
    <!-- VS 标识 -->
    <div _flex="~ center">
      <span _text="3xl" _font="bold" _text="yellow-500">VS</span>
    </div>
    
    <!-- 己方战场 -->
    <div _flex="~ col" _gap="2">
      <div _flex="~" _justify="center" _gap="2">
        <BattleSlot
          v-for="index in 6"
          :key="`ally-${index - 1}`"
          :follower="allyBattlefield[index - 1]"
          :is-enemy="false"
          :slot-index="index - 1"
          :is-drag-over="dragOverSlot === index - 1"
          @dragover.prevent="onDragOver(index - 1)"
          @dragleave="onDragLeave"
          @drop="onDrop(index - 1)"
          @click="selectSlot(index - 1)"
        />
      </div>
      <div _text="green-400" _text="sm" _text="center">己方</div>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { useGameStore } from '@/stores/game'
import { useBattlefieldStore } from '@/stores/battlefield'
import BattleSlot from './BattleSlot.vue'

const gameStore = useGameStore()
const battlefieldStore = useBattlefieldStore()

const allyBattlefield = computed(() => gameStore.currentPlayer?.battlefield || [])
const enemyBattlefield = computed(() => gameStore.opponentPreview?.battlefield || [])

const dragOverSlot = ref<number>(-1)

function onDragOver(slotIndex: number) {
  if (!gameStore.canPosition) return
  dragOverSlot.value = slotIndex
}

function onDragLeave() {
  dragOverSlot.value = -1
}

function onDrop(targetSlot: number) {
  dragOverSlot.value = -1
  battlefieldStore.placeOnBattlefield(targetSlot)
}

function selectSlot(slotIndex: number) {
  const follower = allyBattlefield.value[slotIndex]
  if (follower) {
    battlefieldStore.selectFollower(follower)
  }
}
</script>
```

```vue
import { useGameStore } from '@/stores/game'
import { useBattlefieldStore } from '@/stores/battlefield'

  follower: FollowerInstance | null
  isEnemy: boolean
  slotIndex: number
  isDragOver?: boolean
}>()

  dragover: [event: DragEvent]
  dragleave: [event: DragEvent]
  drop: [event: DragEvent]
  click: []
}>()

const gameStore = useGameStore()
const battlefieldStore = useBattlefieldStore()

const visibleKeywords = computed(() => {
  if (!props.follower) return []
  return props.follower.keywords.filter(k => k.visible)
})

const slotClasses = computed(() => {
  const classes = []
  
  if (props.isEnemy) {
    classes.push('border-red-900', 'opacity-80')
  } else {
    classes.push('border-green-900')
    if (props.isDragOver) {
      classes.push('border-green-500', 'bg-green-900/20')
    }
    if (gameStore.canPosition) {
      classes.push('cursor-pointer')
    }
  }
  
  return classes
})

function returnToHand() {
  battlefieldStore.returnToHand(props.slotIndex)
}
</script>
```

## 3. 拖拽实现

```typescript
// packages/web/src/composables/useDragAndDrop.ts

import { ref } from 'vue'
import type { FollowerInstance } from '@shared/types'

type DragSource = 'hand' | 'battlefield' | 'shop'

export function useDragAndDrop() {
  const draggedItem = ref<{
    follower: FollowerInstance
    source: DragSource
    index: number
  } | null>(null)
  
  const isDragging = ref(false)
  
  function startDrag(
    follower: FollowerInstance,
    source: DragSource,
    index: number
  ) {
    draggedItem.value = { follower, source, index }
    isDragging.value = true
    
    // 设置拖拽数据
    if (event instanceof DragEvent) {
      event.dataTransfer?.setData('application/json', JSON.stringify({
        instanceId: follower.instanceId,
        source,
        index,
      }))
      event.dataTransfer!.effectAllowed = 'move'
    }
  }
  
  function endDrag() {
    draggedItem.value = null
    isDragging.value = false
  }
  
  function handleDrop(
    targetIndex: number,
    targetSource: DragSource,
    callback: (result: {
      follower: FollowerInstance
      fromSource: DragSource
      fromIndex: number
      toIndex: number
    }) => void
  ) {
    if (!draggedItem.value) return
    
    callback({
      follower: draggedItem.value.follower,
      fromSource: draggedItem.value.source,
      fromIndex: draggedItem.value.index,
      toIndex: targetIndex,
    })
    
    endDrag()
  }
  
  return {
    draggedItem,
    isDragging,
    startDrag,
    endDrag,
    handleDrop,
  }
}
```

## 5. 战斗位置工具

```typescript
// packages/web/src/utils/position.ts

import type { BattlePosition } from '@shared/types'

// 槽位索引转坐标
export function slotToPosition(slotIndex: number): BattlePosition {
  return {
    side: 'ally',
    x: slotIndex % 3,
    y: Math.floor(slotIndex / 3),
  }
}

// 坐标转槽位索引
export function positionToSlot(position: BattlePosition): number {
  return position.y * 3 + position.x
}

// 获取敌方镜像位置
export function getEnemyMirrorPosition(allyPosition: BattlePosition): BattlePosition {
  return {
    side: 'enemy',
    x: allyPosition.x,
    y: allyPosition.y,
  }
}

// 获取相邻位置
export function getAdjacentPositions(position: BattlePosition): BattlePosition[] {
  const adjacent: BattlePosition[] = []
  const { x, y } = position
  
  // 上下左右
  if (x > 0) adjacent.push({ side: position.side, x: x - 1, y })
  if (x < 2) adjacent.push({ side: position.side, x: x + 1, y })
  if (y > 0) adjacent.push({ side: position.side, x, y: y - 1 })
  if (y < 1) adjacent.push({ side: position.side, x, y: y + 1 })
  
  return adjacent
}

// 获取前排位置
export function getFrontRowPositions(side: 'ally' | 'enemy'): BattlePosition[] {
  return [
    { side, x: 0, y: 0 },
    { side, x: 1, y: 0 },
    { side, x: 2, y: 0 },
  ]
}

// 获取后排位置
export function getBackRowPositions(side: 'ally' | 'enemy'): BattlePosition[] {
  return [
    { side, x: 0, y: 1 },
    { side, x: 1, y: 1 },
    { side, x: 2, y: 1 },
  ]
}
```

# 02.4 基础战斗

## 目标

实现拼点机制、攻击顺序计算、伤害计算和胜负判定。

## 核心机制（已修改）

```
🔄 伤害机制变更:

原版（互相伤害）:
随从A(攻击5) 攻击 随从B(攻击3)
结果: A生命=10-3=7, B生命=10-5=5

修改版（单向伤害）:
随从A(攻击5) 攻击 随从B(攻击3)
结果: A生命=10(不变), B生命=10-5=5

反击词条是唯一反伤方式:
随从B(攻击3, 有反击) 被随从A攻击
结果: A生命=10-3=7(反击伤害), B生命=10-5=5
```

## 1. 战斗流程

```
战斗阶段流程:

1. 拼点阶段 (Dice Roll)
   ├── 双方掷骰子 (1-6)
   ├── 加上英雄修正
   └── 决定先攻方

2. 战斗进行阶段 (Fighting)
   ├── 先攻方第一只随从攻击
   │   └── 选择目标 → 造成伤害 → 触发词条
   ├── 后攻方第一只随从攻击
   │   └── 同上
   └── 循环直到一方全部阵亡

3. 结果结算阶段 (Result)
   ├── 计算存活随从数
   ├── 计算对英雄的伤害
   └── 确定胜负
```

## 2. 战斗服务 (combatService.ts)

```typescript
// packages/shared/src/services/combatService.ts

import type {
  FollowerInstance,
  Battlefield,
  CombatResult,
  AttackEvent,
  DiceRollResult,
} from '../types'

export class CombatService {
  // 拼点
  rollDice(
    playerAId: string,
    playerBId: string,
    modifierA: number = 0,
    modifierB: number = 0
  ): DiceRollResult {
    const rollA = Math.floor(Math.random() * 6) + 1
    const rollB = Math.floor(Math.random() * 6) + 1
    
    const finalA = rollA + modifierA
    const finalB = rollB + modifierB
    
    return {
      playerA: {
        playerId: playerAId,
        value: rollA,
        modifier: modifierA,
        finalValue: finalA,
      },
      playerB: {
        playerId: playerBId,
        value: rollB,
        modifier: modifierB,
        finalValue: finalB,
      },
      firstAttacker: finalA >= finalB ? playerAId : playerBId,
    }
  }

  // 执行完整战斗
  executeCombat(
    battlefieldA: (FollowerInstance | null)[],
    battlefieldB: (FollowerInstance | null)[],
    firstAttacker: 'A' | 'B'
  ): CombatResult {
    const events: AttackEvent[] = []
    let round = 0
    let currentAttacker: 'A' | 'B' = firstAttacker

    // 创建战场副本（不影响原数据）
    const allyField = this.cloneBattlefield(battlefieldA)
    const enemyField = this.cloneBattlefield(battlefieldB)

    while (round < 100) { // 防止无限循环
      round++
      
      const attackerField = currentAttacker === 'A' ? allyField : enemyField
      const defenderField = currentAttacker === 'A' ? enemyField : allyField
      
      // 获取存活的攻击者
      const aliveAttackers = this.getAliveFollowers(attackerField)
      if (aliveAttackers.length === 0) break
      
      // 按位置顺序选择攻击者
      const attacker = this.selectNextAttacker(attackerField, round)
      if (!attacker) break
      
      // 获取存活的防御者
      const aliveDefenders = this.getAliveFollowers(defenderField)
      if (aliveDefenders.length === 0) break
      
      // 选择目标
      const target = this.selectTarget(attacker, defenderField)
      if (!target) break
      
      // 执行攻击
      const event = this.performAttack(attacker, target, currentAttacker === 'A' ? 'ally' : 'enemy')
      events.push(event)
      
      // 切换攻击方
      currentAttacker = currentAttacker === 'A' ? 'B' : 'A'
    }

    // 计算结果
    const allySurvivors = this.getAliveFollowers(allyField).length
    const enemySurvivors = this.getAliveFollowers(enemyField).length
    
    const isWin = allySurvivors > enemySurvivors
    const damageDealt = isWin ? this.calculateDamage(allyField) : 0

    return {
      isWin,
      damageDealt,
      rounds: round,
      events,
    }
  }

  // 执行单次攻击
  private performAttack(
    attacker: FollowerInstance,
    target: FollowerInstance,
    attackerSide: 'ally' | 'enemy'
  ): AttackEvent {
    const damage = attacker.currentAttack
    const isCritical = false // TODO: 暴击机制

    // 🔄 修改：攻击方只造成伤害，不受到反伤
    // 目标受到伤害
    let actualDamage = damage
    
    // 护盾吸收
    if (target.currentShield > 0) {
      const shieldAbsorb = Math.min(target.currentShield, actualDamage)
      target.currentShield -= shieldAbsorb
      actualDamage -= shieldAbsorb
    }
    
    // 扣除生命
    target.currentHealth -= actualDamage

    // 触发词条
    const triggeredKeywords: Keyword[] = []

    // 攻击时触发
    const onAttackKeywords = attacker.keywords.filter(k => k.trigger === TriggerTiming.ON_ATTACK)
    for (const keyword of onAttackKeywords) {
      this.triggerKeyword(keyword, attacker, target)
      triggeredKeywords.push(keyword)
    }

    // 命中时触发
    const onHitKeywords = attacker.keywords.filter(k => k.trigger === TriggerTiming.ON_HIT)
    for (const keyword of onHitKeywords) {
      this.triggerKeyword(keyword, attacker, target)
      triggeredKeywords.push(keyword)
    }

    // 🔄 修改：只有在目标有反击词条时才对攻击方造成伤害
    const counterKeywords = target.keywords.filter(k => k.trigger === TriggerTiming.ON_DAMAGE)
    for (const keyword of counterKeywords) {
      const counterDamage = target.currentAttack
      attacker.currentHealth -= counterDamage
      triggeredKeywords.push(keyword)
    }

    // 受击时触发
    const onDamageKeywords = target.keywords.filter(k => k.trigger === TriggerTiming.ON_DAMAGE && k.name !== '反击')
    for (const keyword of onDamageKeywords) {
      this.triggerKeyword(keyword, target, attacker)
      triggeredKeywords.push(keyword)
    }

    // 检查击杀
    if (target.currentHealth <= 0) {
      // 击杀触发
      const onKillKeywords = attacker.keywords.filter(k => k.trigger === TriggerTiming.ON_KILL)
      for (const keyword of onKillKeywords) {
        this.triggerKeyword(keyword, attacker, target)
        triggeredKeywords.push(keyword)
      }

      // 遗言触发
      const deathrattleKeywords = target.keywords.filter(k => k.trigger === TriggerTiming.ON_DEATH)
      for (const keyword of deathrattleKeywords) {
        this.triggerKeyword(keyword, target, attacker)
        triggeredKeywords.push(keyword)
      }
    }

    // 检查攻击者死亡（反击导致）
    if (attacker.currentHealth <= 0) {
      const attackerDeathrattle = attacker.keywords.filter(k => k.trigger === TriggerTiming.ON_DEATH)
      for (const keyword of attackerDeathrattle) {
        this.triggerKeyword(keyword, attacker, target)
        triggeredKeywords.push(keyword)
      }
    }

    return {
      attacker,
      target,
      damage,
      isCritical,
      triggeredKeywords,
    }
  }

  // 选择目标
  private selectTarget(
    attacker: FollowerInstance,
    defenderField: (FollowerInstance | null)[]
  ): FollowerInstance | null {
    const aliveDefenders = this.getAliveFollowers(defenderField)
    if (aliveDefenders.length === 0) return null

    // 优先攻击有嘲讽的随从
    const taunters = aliveDefenders.filter(f => 
      f.keywords.some(k => k.name === '嘲讽')
    )
    if (taunters.length > 0) {
      return taunters[0] // 选择第一个嘲讽目标
    }

    // 随机选择目标
    return aliveDefenders[Math.floor(Math.random() * aliveDefenders.length)]
  }

  // 选择下一个攻击者
  private selectNextAttacker(
    field: (FollowerInstance | null)[],
    round: number
  ): FollowerInstance | null {
    const alive = this.getAliveFollowers(field)
    if (alive.length === 0) return null

    // 先手词条优先
    const firstStrikers = alive.filter(f =>
      f.keywords.some(k => k.trigger === TriggerTiming.ON_FIRST_STRIKE && !f.hasAttackedThisRound)
    )
    if (firstStrikers.length > 0) {
      const attacker = firstStrikers[0]
      attacker.hasAttackedThisRound = true
      return attacker
    }

    // 按位置顺序攻击
    for (const follower of field) {
      if (follower && follower.currentHealth > 0 && !follower.hasAttackedThisRound) {
        follower.hasAttackedThisRound = true
        return follower
      }
    }

    // 重置攻击标记，开始下一轮
    for (const follower of field) {
      if (follower) follower.hasAttackedThisRound = false
    }

    // 再次尝试
    for (const follower of field) {
      if (follower && follower.currentHealth > 0 && !follower.hasAttackedThisRound) {
        follower.hasAttackedThisRound = true
        return follower
      }
    }

    return null
  }

  // 获取存活的随从
  private getAliveFollowers(field: (FollowerInstance | null)[]): FollowerInstance[] {
    return field.filter((f): f is FollowerInstance => 
      f !== null && f.currentHealth > 0
    )
  }

  // 计算伤害（根据存活随从）
  private calculateDamage(field: (FollowerInstance | null)[]): number {
    const survivors = this.getAliveFollowers(field)
    // 每个存活随从造成等于其等级的伤害
    return survivors.reduce((total, f) => total + f.level, 0)
  }

  // 克隆战场
  private cloneBattlefield(field: (FollowerInstance | null)[]): (FollowerInstance | null)[] {
    return field.map(f => f ? { ...f, hasAttackedThisRound: false } : null)
  }

  // 触发词条效果
  private triggerKeyword(keyword: Keyword, source: FollowerInstance, target: FollowerInstance): void {
    // TODO: 实现各种效果
    console.log(`触发词条: ${keyword.name}`, { source: source.name, target: target.name })
  }
}

export const combatService = new CombatService()
```

import { ref, computed } from 'vue'

const displayOpponentValue = computed(() => isRolling.value ? '?' : opponentValue.value)

const opponentFinalValue = computed(() => opponentValue.value + opponentModifier.value)

  const result = combatService.rollDice(
    gameStore.currentPlayer!.id,
    gameStore.opponentPreview!.id,
    playerModifier.value,
    opponentModifier.value

  // 动画效果

  const interval = setInterval(() => {
    playerValue.value = Math.floor(Math.random() * 6) + 1
    opponentValue.value = Math.floor(Math.random() * 6) + 1
    rollCount++
    
    if (rollCount >= 10) {
      clearInterval(interval)
      playerValue.value = result.playerA.value
      opponentValue.value = result.playerB.value
      isRolling.value = false
    }
  }, 100)
}

  showDiceRoll.value = false
  gameStore.updatePhase('combat', 'fighting')
}

          造成 {{ currentEvent.damage }} 点伤害
        </div>
      </div>

      <!-- 己方战场 -->
      <div _flex="~" _justify="center" _gap="2">
        <CombatFollower
          v-for="(follower, index) in allyField"
          :key="`ally-${index}`"
          :follower="follower"
          :is-attacking="isAttacking(follower, 'ally')"
          :is-being-attacked="isBeingAttacked(follower)"
          :damage="getDamage(follower)"
        />
      </div>
    </div>

    <!-- 战斗结果 -->
    <CombatResultModal
      v-if="showResult"
      :result="combatResult"
      @close="onResultClose"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import { useGameStore } from '@/stores/game'
import { combatService } from '@shared/services/combatService'
import CombatFollower from './CombatFollower.vue'
import CombatResultModal from './CombatResultModal.vue'

const gameStore = useGameStore()

const allyField = ref(gameStore.currentPlayer?.battlefield || [])
const enemyField = ref(gameStore.opponentPreview?.battlefield || [])
const currentRound = ref(1)
const currentEvent = ref<AttackEvent | null>(null)
const showResult = ref(false)
const combatResult = ref<CombatResult | null>(null)

const isAttacking = (follower: FollowerInstance | null, side: 'ally' | 'enemy') => {
  if (!follower || !currentEvent.value) return false
  return currentEvent.value.attacker.instanceId === follower.instanceId
}

const isBeingAttacked = (follower: FollowerInstance | null) => {
  if (!follower || !currentEvent.value) return false
  return currentEvent.value.target.instanceId === follower.instanceId
}

const getDamage = (follower: FollowerInstance | null) => {
  if (!follower || !currentEvent.value) return 0
  if (currentEvent.value.target.instanceId === follower.instanceId) {
    return currentEvent.value.damage
  }
  return 0
}

// 播放战斗动画
async function playCombat() {
  const result = combatService.executeCombat(
    allyField.value,
    enemyField.value,
    gameStore.isMyTurn ? 'A' : 'B'
  )

  // 逐个播放攻击事件
  for (let i = 0; i < result.events.length; i++) {
    currentEvent.value = result.events[i]
    await sleep(1000) // 等待动画
    
    // 更新生命值
    updateHealth(currentEvent.value)
    await sleep(500)
  }

  combatResult.value = result
  showResult.value = true
  gameStore.setCombatResult(result)
}

function updateHealth(event: AttackEvent) {
  // 更新目标生命值
  event.target.currentHealth -= event.damage
  // 处理反击伤害
  if (event.triggeredKeywords.some(k => k.name === '反击')) {
    event.attacker.currentHealth -= event.target.currentAttack
  }
}

function onResultClose() {
  showResult.value = false
  gameStore.updatePhase('recruit', 'shopping')
}

function sleep(ms: number): Promise<void> {
  return new Promise(resolve => setTimeout(resolve, ms))
}

// 自动开始战斗
playCombat()
</script>

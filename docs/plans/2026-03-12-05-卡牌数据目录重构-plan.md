# 卡牌数据目录重构 实现计划

> **给 AI 助手：** 必需子技能：使用 executing-plans 或 subagent-driven-development 实现此计划。选择取决于执行环境偏好。

**目标：** 将卡牌数据重构为"类型/等级/单卡"三级目录结构，优化 AI 读取效率

**架构：** 按卡牌类型（followers/equipment/spells）创建顶级目录，每个类型下按等级（lv1~lv6/lv2~lv6）创建子目录，每个卡牌单独一个 TypeScript 文件

**技术栈：** TypeScript, Bun, Vitest

**设计文档：** `docs/design/2026-03-12-卡牌数据目录重构-design.md`

## 任务 4：创建随从类型索引文件

**文件：**
- 创建：`packages/shared/src/data/followers/index.ts`

**步骤 1：创建类型索引文件**

```typescript
// packages/shared/src/data/followers/index.ts
import type { Follower } from '../types/follower'

import { FOLLOWERS_LV1 } from './lv1'
import { FOLLOWERS_LV2 } from './lv2'
import { FOLLOWERS_LV3 } from './lv3'
import { FOLLOWERS_LV4 } from './lv4'
import { FOLLOWERS_LV5 } from './lv5'
import { FOLLOWERS_LV6 } from './lv6'

// 按等级导出
export { FOLLOWERS_LV1, FOLLOWERS_LV2, FOLLOWERS_LV3, FOLLOWERS_LV4, FOLLOWERS_LV5, FOLLOWERS_LV6 }

// 导出单卡
export * from './lv1'
export * from './lv2'
export * from './lv3'
export * from './lv4'
export * from './lv5'
export * from './lv6'

// 所有随从
export const ALL_FOLLOWERS: Follower[] = [
  ...FOLLOWERS_LV1,
  ...FOLLOWERS_LV2,
  ...FOLLOWERS_LV3,
  ...FOLLOWERS_LV4,
  ...FOLLOWERS_LV5,
  ...FOLLOWERS_LV6
]

// 按 ID 查找
export const FOLLOWER_MAP: Record<string, Follower> = Object.fromEntries(
  ALL_FOLLOWERS.map(f => [f.id, f])
)

// 兼容旧接口
export const FOLLOWERS: Record<string, Follower> = FOLLOWER_MAP

export function getFollower(id: string): Follower | undefined {
  return FOLLOWERS[id]
}

export function getAllFollowers(): Follower[] {
  return ALL_FOLLOWERS
}

export function getFollowersByLevel(level: 1 | 2 | 3 | 4 | 5 | 6): Follower[] {
  switch (level) {
    case 1: return FOLLOWERS_LV1
    case 2: return FOLLOWERS_LV2
    case 3: return FOLLOWERS_LV3
    case 4: return FOLLOWERS_LV4
    case 5: return FOLLOWERS_LV5
    case 6: return FOLLOWERS_LV6
  }
}
```

## 1. 数据库 Schema (schema.ts)

### 新增/修改表结构

```typescript
// packages/server/src/db/schema.ts

import { pgTable, uuid, varchar, integer, boolean, jsonb, timestamp, text } from 'drizzle-orm/pg-core'

// ==================== 用户表 ====================
export const users = pgTable('users', {
  id: uuid('id').primaryKey().defaultRandom(),
  username: varchar('username', { length: 255 }).notNull().unique(),
  email: varchar('email', { length: 255 }).unique(),
  passwordHash: varchar('password_hash', { length: 255 }),
  avatarUrl: varchar('avatar_url', { length: 500 }),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
})

// ==================== 游戏房间表 ====================
export const gameRooms = pgTable('game_rooms', {
  id: uuid('id').primaryKey().defaultRandom(),
  name: varchar('name', { length: 100 }).notNull(),
  hostId: uuid('host_id').references(() => users.id).notNull(),
  maxPlayers: integer('max_players').notNull().default(8),
  status: varchar('status', { length: 20 }).notNull().default('waiting'), // waiting, playing, ended
  createdAt: timestamp('created_at').defaultNow(),
  startedAt: timestamp('started_at'),
  endedAt: timestamp('ended_at'),
  config: jsonb('config').default({}), // 房间配置
})

// ==================== 房间玩家关联表 ====================
export const roomPlayers = pgTable('room_players', {
  id: uuid('id').primaryKey().defaultRandom(),
  roomId: uuid('room_id').references(() => gameRooms.id).notNull(),
  userId: uuid('user_id').references(() => users.id).notNull(),
  slot: integer('slot').notNull(), // 玩家位置 0-7
  isReady: boolean('is_ready').default(false),
  isHost: boolean('is_host').default(false),
  joinedAt: timestamp('joined_at').defaultNow(),
})

// ==================== 游戏会话表 ====================
export const gameSessions = pgTable('game_sessions', {
  id: uuid('id').primaryKey().defaultRandom(),
  roomId: uuid('room_id').references(() => gameRooms.id).notNull(),
  currentRound: integer('current_round').default(1),
  phase: varchar('phase', { length: 20 }).notNull().default('lobby'), // lobby, recruit, combat, settlement, ended
  subPhase: varchar('sub_phase', { length: 20 }), // shopping, positioning, fighting, result
  turnTimer: integer('turn_timer').default(60),
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
  endedAt: timestamp('ended_at'),
  winnerId: uuid('winner_id').references(() => users.id),
})

// ==================== 玩家游戏状态表 ====================
export const playerGameStates = pgTable('player_game_states', {
  id: uuid('id').primaryKey().defaultRandom(),
  sessionId: uuid('session_id').references(() => gameSessions.id).notNull(),
  userId: uuid('user_id').references(() => users.id).notNull(),
  
  // 基础状态
  health: integer('health').notNull().default(40),
  maxHealth: integer('max_health').notNull().default(40),
  gold: integer('gold').notNull().default(3),
  level: integer('level').notNull().default(1),
  experience: integer('experience').default(0),
  
  // 英雄
  heroId: varchar('hero_id', { length: 50 }),
  
  // 复杂数据结构用 JSONB
  battlefield: jsonb('battlefield').default([]), // FollowerInstance[]
  hand: jsonb('hand').default([]), // FollowerInstance[]
  shop: jsonb('shop').default([]), // Follower[]
  
  // 状态
  isAlive: boolean('is_alive').default(true),
  rank: integer('rank'),
  
  createdAt: timestamp('created_at').defaultNow(),
  updatedAt: timestamp('updated_at').defaultNow(),
})

// ==================== 战斗记录表 ====================
export const combatRecords = pgTable('combat_records', {
  id: uuid('id').primaryKey().defaultRandom(),
  sessionId: uuid('session_id').references(() => gameSessions.id).notNull(),
  round: integer('round').notNull(),
  playerAId: uuid('player_a_id').references(() => users.id).notNull(),
  playerBId: uuid('player_b_id').references(() => users.id).notNull(),
  
  // 战斗结果
  winnerId: uuid('winner_id').references(() => users.id),
  damageDealt: integer('damage_dealt').default(0),
  rounds: integer('rounds').default(0), // 战斗持续回合数
  
  // 完整战斗记录（用于回放）
  combatLog: jsonb('combat_log').default([]),
  
  createdAt: timestamp('created_at').defaultNow(),
})

// ==================== 玩家统计表 ====================
export const playerStats = pgTable('player_stats', {
  id: uuid('id').primaryKey().defaultRandom(),
  userId: uuid('user_id').references(() => users.id).notNull().unique(),
  
  gamesPlayed: integer('games_played').default(0),
  gamesWon: integer('games_won').default(0),
  gamesTop4: integer('games_top4').default(0),
  totalDamageDealt: integer('total_damage_dealt').default(0),
  totalGoldEarned: integer('total_gold_earned').default(0),
  highestRank: integer('highest_rank'),
  
  updatedAt: timestamp('updated_at').defaultNow(),
})
```

## 3. 房间服务 (roomService.ts)

```typescript
// packages/server/src/services/roomService.ts

import { eq, and } from 'drizzle-orm'
import { db, gameRooms, roomPlayers } from '../db'
import type { GameRoom, Player } from '@shared/types'

export class RoomService {
  // 创建房间
  async createRoom(hostId: string, name: string, maxPlayers: number = 8): Promise<GameRoom> {
    const [room] = await db
      .insert(gameRooms)
      .values({
        name,
        hostId,
        maxPlayers,
        status: 'waiting',
      })
      .returning()
    
    // 房主自动加入
    await this.joinRoom(room.id, hostId, true)
    
    return this.mapToGameRoom(room)
  }
  
  // 加入房间
  async joinRoom(roomId: string, userId: string, isHost: boolean = false): Promise<void> {
    // 检查是否已满
    const playerCount = await db
      .select({ count: roomPlayers.id })
      .from(roomPlayers)
      .where(eq(roomPlayers.roomId, roomId))
    
    if (playerCount.length >= 8) {
      throw new Error('Room is full')
    }
    
    // 查找空位
    const existingSlots = await db
      .select({ slot: roomPlayers.slot })
      .from(roomPlayers)
      .where(eq(roomPlayers.roomId, roomId))
    
    const usedSlots = new Set(existingSlots.map(p => p.slot))
    let slot = 0
    while (usedSlots.has(slot) && slot < 8) slot++
    
    await db.insert(roomPlayers).values({
      roomId,
      userId,
      slot,
      isHost,
    })
  }
  
  // 离开房间
  async leaveRoom(roomId: string, userId: string): Promise<void> {
    await db
      .delete(roomPlayers)
      .where(and(
        eq(roomPlayers.roomId, roomId),
        eq(roomPlayers.userId, userId)
      ))
    
    // 检查房间是否为空
    const remaining = await db
      .select({ count: roomPlayers.id })
      .from(roomPlayers)
      .where(eq(roomPlayers.roomId, roomId))
    
    if (remaining.length === 0) {
      // 删除空房间
      await db.delete(gameRooms).where(eq(gameRooms.id, roomId))
    }
  }
  
  // 设置准备状态
  async setReady(roomId: string, userId: string, isReady: boolean): Promise<void> {
    await db
      .update(roomPlayers)
      .set({ isReady })
      .where(and(
        eq(roomPlayers.roomId, roomId),
        eq(roomPlayers.userId, userId)
      ))
  }
  
  // 检查是否可以开始游戏
  async canStartGame(roomId: string): Promise<boolean> {
    const players = await db
      .select({ isReady: roomPlayers.isReady })
      .from(roomPlayers)
      .where(eq(roomPlayers.roomId, roomId))
    
    // 至少2人且全部准备
    return players.length >= 2 && players.every(p => p.isReady)
  }
  
  // 获取房间列表
  async getRoomList(): Promise<GameRoom[]> {
    const rooms = await db
      .select()
      .from(gameRooms)
      .where(eq(gameRooms.status, 'waiting'))
    
    return rooms.map(r => this.mapToGameRoom(r))
  }
  
  // 获取房间详情
  async getRoom(roomId: string): Promise<GameRoom | null> {
    const [room] = await db
      .select()
      .from(gameRooms)
      .where(eq(gameRooms.id, roomId))
    
    if (!room) return null
    
    const players = await db
      .select()
      .from(roomPlayers)
      .where(eq(roomPlayers.roomId, roomId))
    
    return {
      ...this.mapToGameRoom(room),
      players: players.map(p => ({
        id: p.userId,
        slot: p.slot,
        isReady: p.isReady,
        isHost: p.isHost,
      })),
    }
  }
  
  // 映射函数
  private mapToGameRoom(dbRoom: typeof gameRooms.$inferSelect): GameRoom {
    return {
      id: dbRoom.id,
      name: dbRoom.name,
      hostId: dbRoom.hostId,
      maxPlayers: dbRoom.maxPlayers,
      status: dbRoom.status as GameRoom['status'],
      createdAt: dbRoom.createdAt.getTime(),
      startedAt: dbRoom.startedAt?.getTime(),
      endedAt: dbRoom.endedAt?.getTime(),
      players: [],
    }
  }
}

export const roomService = new RoomService()
```

## 4. 游戏状态服务 (gameService.ts)

```typescript
// packages/server/src/services/gameService.ts

import { eq } from 'drizzle-orm'
import { db, gameSessions, playerGameStates, combatRecords } from '../db'
import type { GameSession, Player, CombatResult } from '@shared/types'

export class GameService {
  // 创建游戏会话
  async createSession(roomId: string, playerIds: string[]): Promise<GameSession> {
    const [session] = await db
      .insert(gameSessions)
      .values({
        roomId,
        currentRound: 1,
        phase: 'hero_select',
        turnTimer: 60,
      })
      .returning()
    
    // 初始化玩家状态
    for (const userId of playerIds) {
      await db.insert(playerGameStates).values({
        sessionId: session.id,
        userId,
        health: 40,
        maxHealth: 40,
        gold: 3,
        level: 1,
        battlefield: [],
        hand: [],
        shop: [],
      })
    }
    
    return this.mapToGameSession(session)
  }
  
  // 获取会话
  async getSession(sessionId: string): Promise<GameSession | null> {
    const [session] = await db
      .select()
      .from(gameSessions)
      .where(eq(gameSessions.id, sessionId))
    
    return session ? this.mapToGameSession(session) : null
  }
  
  // 更新游戏阶段
  async updatePhase(
    sessionId: string,
    phase: GameSession['phase'],
    subPhase?: string
  ): Promise<void> {
    await db
      .update(gameSessions)
      .set({
        phase,
        subPhase: subPhase || null,
        updatedAt: new Date(),
      })
      .where(eq(gameSessions.id, sessionId))
  }
  
  // 获取玩家状态
  async getPlayerState(sessionId: string, userId: string): Promise<Player | null> {
    const [state] = await db
      .select()
      .from(playerGameStates)
      .where(and(
        eq(playerGameStates.sessionId, sessionId),
        eq(playerGameStates.userId, userId)
      ))
    
    if (!state) return null
    
    return {
      id: state.userId,
      health: state.health,
      maxHealth: state.maxHealth,
      gold: state.gold,
      level: state.level,
      experience: state.experience,
      battlefield: state.battlefield as Player['battlefield'],
      hand: state.hand as Player['hand'],
      shop: state.shop as Player['shop'],
      isAlive: state.isAlive,
      rank: state.rank || undefined,
    }
  }
  
  // 更新玩家状态
  async updatePlayerState(sessionId: string, userId: string, updates: Partial<Player>): Promise<void> {
    await db
      .update(playerGameStates)
      .set({
        ...updates,
        updatedAt: new Date(),
      })
      .where(and(
        eq(playerGameStates.sessionId, sessionId),
        eq(playerGameStates.userId, userId)
      ))
  }
  
  // 记录战斗
  async recordCombat(
    sessionId: string,
    round: number,
    playerAId: string,
    playerBId: string,
    result: CombatResult
  ): Promise<void> {
    await db.insert(combatRecords).values({
      sessionId,
      round,
      playerAId,
      playerBId,
      winnerId: result.isWin ? playerAId : playerBId,
      damageDealt: result.damageDealt,
      rounds: result.rounds,
      combatLog: result.events,
    })
  }
  
  // 映射函数
  private mapToGameSession(dbSession: typeof gameSessions.$inferSelect): GameSession {
    return {
      id: dbSession.id,
      roomId: dbSession.roomId,
      currentRound: dbSession.currentRound,
      phase: dbSession.phase as GameSession['phase'],
      subPhase: dbSession.subPhase as GameSession['subPhase'],
      turnTimer: dbSession.turnTimer,
      timestamp: dbSession.updatedAt.getTime(),
    }
  }
}

export const gameService = new GameService()
```

## 5. API 路由

```typescript
// packages/server/src/routes/api/rooms.ts

import { Hono } from 'hono'
import { roomService } from '../../services/roomService'
import { authMiddleware } from '../../middleware/auth'

const rooms = new Hono()

// 获取房间列表
rooms.get('/', async (c) => {
  const roomList = await roomService.getRoomList()
  return c.json({
    success: true,
    data: roomList,
    timestamp: Date.now(),
  })
})

// 创建房间
rooms.post('/', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const { name, maxPlayers } = await c.req.json()
  
  const room = await roomService.createRoom(userId, name, maxPlayers)
  return c.json({
    success: true,
    data: room,
    timestamp: Date.now(),
  })
})

// 加入房间
rooms.post('/:id/join', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const roomId = c.req.param('id')
  
  await roomService.joinRoom(roomId, userId)
  return c.json({
    success: true,
    message: 'Joined room successfully',
    timestamp: Date.now(),
  })
})

// 离开房间
rooms.post('/:id/leave', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const roomId = c.req.param('id')
  
  await roomService.leaveRoom(roomId, userId)
  return c.json({
    success: true,
    message: 'Left room successfully',
    timestamp: Date.now(),
  })
})

// 设置准备状态
rooms.post('/:id/ready', authMiddleware, async (c) => {
  const userId = c.get('userId')
  const roomId = c.req.param('id')
  const { isReady } = await c.req.json()
  
  await roomService.setReady(roomId, userId, isReady)
  return c.json({
    success: true,
    message: isReady ? 'Ready' : 'Not ready',
    timestamp: Date.now(),
  })
})

export default rooms
```
